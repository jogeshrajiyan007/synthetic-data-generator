"""
services/rag.py

Conversational RAG + SQL Agent hybrid engine.

Routing logic:
  1. If enable_sql_agent=True and approved Databricks tables exist:
     → Keyword pre-classifier runs first (no LLM call, no ChromaDB lookup)
     → If definite SQL intent: skip RAG retrieval, go straight to SQL agent
     → If ambiguous: run RAG retrieval for context, then SQL agent
     → If NOT_SQL: fall through to standard RAG
  2. Standard RAG: condense → retrieve → answer
  3. Direct (no retrieval): history-aware direct LLM call

Compatible with LangChain >= 0.2.x — no deprecated chains used.
"""

# ── Approved sources block ────────────────────────────────────────

def _build_approved_sources_block(approved_tables: list[str]) -> str:
    """
    Injected into the RAG system prompt only when routing reaches the RAG path.
    This only happens for genuine governance/doc questions (NOT_SQL intent).
    """
    if not approved_tables:
        return ""
    lines = "\n".join(f"  - {t}" for t in approved_tables)
    return f"""
APPROVED DATABRICKS TABLES (available for live SQL queries via the SQL Agent):
<approved_sources>
{lines}
</approved_sources>

NOTE: You are answering a governance or documentation question — NOT a live data query.
If the user is asking to retrieve actual rows or run analytics, tell them:
"To query live data, ask a specific data question like \
'Show me the first 10 rows of claims_casualty_bronze' and the SQL Agent will execute it."
Do NOT suggest users run SQL manually in their Databricks environment.
"""


# ── Retriever ─────────────────────────────────────────────────────

def _build_retriever(vectordb: Chroma, k: int):
    """MMR retriever with multi-source representation guarantee."""
    base = vectordb.as_retriever(
        search_type="mmr",
        search_kwargs={"k": k * 2, "fetch_k": k * 6},
    )

    def multi_source_retrieve(query: str) -> list[Document]:
        docs = base.invoke(query)
        seen: dict[str, Document] = {}
        for doc in docs:
            src = doc.metadata.get("source_file", "unknown")
            if src not in seen:
                seen[src] = doc
        if len(seen) < 2:
            try:
                all_srcs = list({
                    d.metadata.get("source_file", "unknown")
                    for d in vectordb.similarity_search("data governance summary", k=50)
                })
                for src in all_srcs:
                    if src not in seen:
                        results = vectordb.similarity_search(
                            query, k=2, filter={"source_file": src}
                        )
                        if results:
                            seen[src] = results[0]
            except Exception:
                pass
        truncated: list[Document] = []
        for doc in seen.values():
            if len(doc.page_content) > 1500:
                doc = Document(
                    page_content=doc.page_content[:1500] + "\n[...truncated]",
                    metadata=doc.metadata,
                )
            truncated.append(doc)
        return truncated

    return multi_source_retrieve


def _format_docs(docs: list[Document]) -> str:
    parts = []
    for doc in docs:
        src = doc.metadata.get("source_file", "unknown")
        parts.append(f"[Source: {src}]\n{doc.page_content}")
    return "\n\n---\n\n".join(parts)


# ── Token extraction ──────────────────────────────────────────────

def _extract_tokens(llm_response) -> dict:
    try:
        usage = getattr(llm_response, "usage_metadata", None) or {}
        return {
            "prompt_tokens"    : int(usage.get("input_tokens",  0)),
            "completion_tokens": int(usage.get("output_tokens", 0)),
            "total_tokens"     : int(usage.get("total_tokens",  0)),
        }
    except Exception:
        return {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}


# ── Main query function ───────────────────────────────────────────

def query_governance(
    question: str,
    vectordb: Chroma,
    knowledge_set: str = "default",
    session_id: str = "",
    use_rag: bool = True,
    k: int = RETRIEVAL_K,
    window_k: int = DEFAULT_WINDOW_K,
    enable_sql_agent: bool = True,
    approved_tables: Optional[list[str]] = None,
    connection_id: Optional[str] = None,
) -> dict:
    """
    Hybrid query engine: SQL Agent + RAG.

    Routing:
      1. If enable_sql_agent=True and approved_tables exist:
         a. Keyword pre-classifier runs instantly (no LLM, no ChromaDB)
         b. Definite SQL intent → skip RAG retrieval, run SQL agent directly
         c. Ambiguous → run RAG retrieval for context, then SQL agent
         d. NOT_SQL → fall through to standard RAG
      2. Standard LCEL RAG: condense → retrieve → answer
      3. Direct LLM: history-aware, no retrieval
    """
    start   = time.time()
    run_id  = str(uuid.uuid4())
    answer  = ""
    llm     = get_llm()
    parser  = StrOutputParser()

    if not session_id:
        session_id = knowledge_set

    approved_tables = approved_tables or []

    # ── Sync approved sources to SQLite ──────────────────────────
    if approved_tables and connection_id:
        try:
            from services.databricks.approved_sources import sync_approved_sources
            sync_approved_sources(
                session_id=session_id,
                knowledge_set=knowledge_set,
                connection_id=connection_id,
                tables=approved_tables,
            )
        except Exception:
            pass

    # ── Load conversation history ─────────────────────────────────
    chat_history: list[BaseMessage] = load_history_as_messages(
        session_id, knowledge_set, window_k
    )

    # ── SQL Agent routing ─────────────────────────────────────────
    # Requires BOTH:
    #   1. enable_sql_agent=True (frontend toggle must be ON)
    #   2. approved_tables non-empty (user must have approved at least one table)
    if enable_sql_agent and approved_tables:
        sql_agent_error = None
        try:
            from services.databricks.sql_agent import run_sql_agent, _keyword_intent, INTENT_PROMPT
            from langchain_core.output_parsers import StrOutputParser as _StrParser

            approved_list_str = "\n".join(f"  - {t}" for t in approved_tables)

            # ── Stage 1: Keyword pre-classifier ──────────────────────
            # Returns a SQL intent string if confident, None if ambiguous.
            quick_intent = _keyword_intent(question, approved_tables)

            # ── Stage 2: Resolve ambiguous questions via LLM ─────────
            # Only call the LLM if the keyword classifier was uncertain.
            if quick_intent is None:
                try:
                    _parser = _StrParser()
                    _intent_chain = INTENT_PROMPT | llm | _parser
                    _llm_intent = _intent_chain.invoke({
                        "approved_sources_list": approved_list_str,
                        "question": question,
                    }).strip().upper()
                    resolved_intent = _llm_intent if _llm_intent in {
                        "SIMPLE_LOOKUP", "AGGREGATION", "COMPARISON",
                        "JOIN_OPERATION", "MULTI_STEP", "CREATE_OBJECT",
                        "OUT_OF_SCOPE", "NOT_SQL"
                    } else "NOT_SQL"
                except Exception:
                    resolved_intent = "NOT_SQL"
            else:
                resolved_intent = quick_intent

            # ── Stage 3: Gate — only proceed to agent for SQL intents ─
            # NOT_SQL and OUT_OF_SCOPE_DOC both fall through to RAG.
            # OUT_OF_SCOPE (unapproved table) IS a SQL intent — show the
            # "not in approved sources" message rather than hallucinating.
            SQL_GATE_INTENTS = {
                "SIMPLE_LOOKUP", "AGGREGATION", "COMPARISON",
                "JOIN_OPERATION", "MULTI_STEP", "CREATE_OBJECT",
            }

            if resolved_intent not in SQL_GATE_INTENTS:
                # NOT_SQL or unrecognised → fall through to RAG below
                raise StopIteration  # use as a clean non-exception exit

            # Definite SQL — now fetch RAG context for schema/governance overlay
            is_definite_sql = quick_intent is not None
            rag_context        = ""
            governance_context = ""
            schema_context     = ""
            rag_sources: list[str] = []
            rag_chunks         = 0

            if not is_definite_sql:
                try:
                    retriever_fn = _build_retriever(vectordb, k)
                    source_docs  = retriever_fn(question)
                    rag_context  = _format_docs(source_docs)
                    rag_sources  = list({
                        doc.metadata.get("source_file", "unknown")
                        for doc in source_docs
                    })
                    rag_chunks = len(source_docs)
                    gov_docs = [
                        d for d in source_docs
                        if d.metadata.get("source_type") == "databricks"
                        or any(t in d.page_content.lower() for t in
                               ["pii", "sensitivity", "owner", "gdpr", "solvency"])
                    ]
                    governance_context = _format_docs(gov_docs)
                    schema_docs = [
                        d for d in source_docs
                        if d.metadata.get("file_type") in
                        ("databricks_schema", "databricks_stats")
                    ]
                    schema_context = _format_docs(schema_docs)
                except Exception:
                    pass

            agent_result = run_sql_agent(
                question                 = question,
                session_id               = session_id,
                knowledge_set            = knowledge_set,
                chat_history             = chat_history,
                rag_context              = rag_context,
                governance_context       = governance_context,
                schema_context           = schema_context,
                approved_tables_override = approved_tables,
                connection_id_override   = connection_id,
                intent_override          = resolved_intent,
            )

            intent = agent_result.get("intent", "NOT_SQL")

            # All SQL-related intents return agent result.
            # Only NOT_SQL falls through to standard RAG below.
            SQL_INTENTS = {
                "SIMPLE_LOOKUP", "AGGREGATION", "COMPARISON",
                "JOIN_OPERATION", "MULTI_STEP", "CREATE_OBJECT", "OUT_OF_SCOPE",
            }

            if intent in SQL_INTENTS:
                answer_parts = []
                if agent_result.get("think_aloud"):
                    answer_parts.append(f"_{agent_result['think_aloud']}_\n")
                if agent_result.get("interpretation"):
                    answer_parts.append(agent_result["interpretation"])
                if agent_result.get("governance_note"):
                    answer_parts.append(f"\n\n{agent_result['governance_note']}")
                answer = "\n".join(answer_parts) if answer_parts else "Query executed."

                result = {
                    "run_id"           : run_id,
                    "session_id"       : session_id,
                    "knowledge_set"    : knowledge_set,
                    "question"         : question,
                    "rephrased_question": question,
                    "answer"           : answer,
                    "mode"             : agent_result.get("mode", "sql_agent"),
                    "sources_used"     : rag_sources,
                    "chunks_retrieved" : rag_chunks,
                    "model"            : CHAT_MODEL,
                    "prompt_tokens"    : 0,
                    "completion_tokens": 0,
                    "total_tokens"     : 0,
                    "created_at"       : datetime.now().isoformat(),
                    "duration_seconds" : round(time.time() - start, 2),
                    "sql_intent"       : intent,
                    "sql_executed"     : agent_result.get("sql", ""),
                    "sql_row_count"    : agent_result.get("row_count", 0),
                    "sql_results"      : agent_result.get("results", []),
                    "sql_think_aloud"  : agent_result.get("think_aloud", ""),
                    "governance_note"  : agent_result.get("governance_note", ""),
                }

                db_insert("llm_responses", {
                    k2: v for k2, v in result.items()
                    if not isinstance(v, (list, dict)) or k2 == "sources_used"
                })
                save_turn(
                    session_id=session_id, knowledge_set=knowledge_set,
                    human_message=question, ai_message=answer,
                    run_id=run_id, token_count=0,
                )
                return result

            # NOT_SQL intent — fall through to standard RAG below
            # (intentional — no exception raised)

        except StopIteration:
            pass  # Intentional: resolved_intent was NOT_SQL → fall to RAG

        except Exception as e:
            import traceback
            print(f"[SQL AGENT ERROR] {type(e).__name__}: {e}")
            traceback.print_exc()
            sql_agent_error = str(e)

    # ── Standard RAG ─────────────────────────────────────────────
    approved_sources_block = _build_approved_sources_block(approved_tables)

    if use_rag:
        # Condense follow-up questions into standalone questions
        if chat_history:
            condense_chain = CONDENSE_PROMPT | llm | parser
            rephrased      = condense_chain.invoke({
                "chat_history": chat_history,
                "question"    : question,
            })
        else:
            rephrased = question

        retriever_fn = _build_retriever(vectordb, k)
        source_docs  = retriever_fn(rephrased)
        context      = _format_docs(source_docs)
        # Filter sources_used to only include files/tables that are
        # currently active (approved_tables or uploaded files).
        # This prevents removed Databricks tables from showing as sources.
        all_sources = list({
            doc.metadata.get("source_file", "unknown") for doc in source_docs
        })
        if approved_tables:
            # Keep a source if it matches an approved table short name OR
            # is a locally uploaded file (not a databricks source_type)
            approved_short = {t.split(".")[-1].lower() for t in approved_tables}
            sources = [
                s for s in all_sources
                if s.lower() in approved_short
                or not any(
                    doc.metadata.get("source_type") == "databricks"
                    for doc in source_docs
                    if doc.metadata.get("source_file") == s
                )
            ]
        else:
            sources = all_sources

        qa_messages = QA_PROMPT.format_messages(
            approved_sources_block=approved_sources_block,
            context=context,
            chat_history=chat_history,
            question=question,
        )
        llm_response = llm.invoke(qa_messages)
        answer       = llm_response.content
        token_usage  = _extract_tokens(llm_response)

        result = {
            "run_id"           : run_id,
            "session_id"       : session_id,
            "knowledge_set"    : knowledge_set,
            "question"         : question,
            "rephrased_question": rephrased,
            "answer"           : answer,
            "mode"             : "conversational_rag",
            "sources_used"     : sources,
            "chunks_retrieved" : len(source_docs),
            "model"            : CHAT_MODEL,
            "prompt_tokens"    : token_usage["prompt_tokens"],
            "completion_tokens": token_usage["completion_tokens"],
            "total_tokens"     : token_usage["total_tokens"],
            "created_at"       : datetime.now().isoformat(),
            "duration_seconds" : round(time.time() - start, 2),
            "sql_intent"       : None,
            "sql_executed"     : None,
            "sql_row_count"    : None,
            "sql_results"      : None,
            "sql_think_aloud"  : None,
            "governance_note"  : None,
        }

    else:
        # ── Direct LLM (no retrieval) ─────────────────────────────
        direct_prompt = ChatPromptTemplate.from_messages([
            (
                "system",
                "You are a Data & AI Expert for the insurance industry. "
                "You have full memory of this conversation. "
                "Build on previous answers when relevant.\n\n"
                f"{approved_sources_block}",
            ),
            MessagesPlaceholder(variable_name="chat_history"),
            ("human", "{question}"),
        ])
        direct_messages = direct_prompt.format_messages(
            chat_history=chat_history,
            question=question,
        )
        llm_response = llm.invoke(direct_messages)
        answer       = llm_response.content
        token_usage  = _extract_tokens(llm_response)

        result = {
            "run_id"           : run_id,
            "session_id"       : session_id,
            "knowledge_set"    : knowledge_set,
            "question"         : question,
            "rephrased_question": question,
            "answer"           : answer,
            "mode"             : "direct",
            "sources_used"     : [],
            "chunks_retrieved" : 0,
            "model"            : CHAT_MODEL,
            "prompt_tokens"    : token_usage["prompt_tokens"],
            "completion_tokens": token_usage["completion_tokens"],
            "total_tokens"     : token_usage["total_tokens"],
            "created_at"       : datetime.now().isoformat(),
            "duration_seconds" : round(time.time() - start, 2),
            "sql_intent"       : None,
            "sql_executed"     : None,
            "sql_row_count"    : None,
            "sql_results"      : None,
            "sql_think_aloud"  : None,
            "governance_note"  : None,
        }

    # ── Persist ───────────────────────────────────────────────────
    db_insert("llm_responses", {
        k2: v for k2, v in result.items()
        if not isinstance(v, (list, dict)) or k2 == "sources_used"
    })
    save_turn(
        session_id=session_id, knowledge_set=knowledge_set,
        human_message=question, ai_message=answer,
        run_id=run_id, token_count=result.get("total_tokens", 0),
    )
    return result
