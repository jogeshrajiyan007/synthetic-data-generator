from fastapi import APIRouter, HTTPException
from services.rag import query_governance
from services.vectorstore import load_vector_store
from models.schemas import QueryRequest, QueryResponse

router = APIRouter(prefix="/api/query", tags=["Query"])


def _get_vectordb():
    try:
        return load_vector_store()
    except Exception:
        raise HTTPException(
            status_code=503,
            detail="Vector store not initialised. Please call POST /api/ingest first."
        )


@router.post("/", response_model=QueryResponse)
async def query_endpoint(request: QueryRequest):
    """
    Send a governance question. Returns a structured answer with sources.
    """
    vectordb = _get_vectordb()
    result   = query_governance(
        question=request.question,
        vectordb=vectordb,
        use_rag=request.use_rag,
        k=request.k
    )
    return QueryResponse(**result)


@router.post("/batch", response_model=list[QueryResponse])
async def batch_query(requests: list[QueryRequest]):
    """
    Run multiple queries in sequence. Useful for running the default
    governance assessment set programmatically.
    """
    if len(requests) > 20:
        raise HTTPException(status_code=400, detail="Max 20 queries per batch.")

    vectordb = _get_vectordb()
    results  = []
    for req in requests:
        result = query_governance(
            question=req.question,
            vectordb=vectordb,
            use_rag=req.use_rag,
            k=req.k
        )
        results.append(QueryResponse(**result))
    return results
