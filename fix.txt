import re
import uuid
import json
from datetime import datetime
from langchain_core.documents import Document
from langchain_text_splitters import RecursiveCharacterTextSplitter
from db import db_insert_batch
from config import (
    CHUNK_SIZE, CHUNK_OVERLAP,
    PII_SIGNALS, SENSITIVITY_TIERS,
    DEPARTMENT_SIGNALS, GOVERNANCE_THEMES
)


def _detect_sensitivity(text: str) -> str:
    lower = text.lower()
    for tier, signals in SENSITIVITY_TIERS.items():
        if any(s in lower for s in signals):
            return tier
    return "Public"


def _detect_department(text: str) -> str:
    lower  = text.lower()
    scores = {
        dept: sum(1 for s in signals if s in lower)
        for dept, signals in DEPARTMENT_SIGNALS.items()
    }
    best = max(scores, key=scores.get)
    return best if scores[best] > 0 else "General"


def _extract_keywords(text: str, top_n: int = 8) -> list[str]:
    words = re.findall(r"\b[a-zA-Z]{4,}\b", text.lower())
    stop  = {"that", "this", "with", "from", "have", "been", "will",
             "they", "their", "which", "were", "also", "more", "than"}
    counts: dict[str, int] = {}
    for w in words:
        if w not in stop:
            counts[w] = counts.get(w, 0) + 1
    return sorted(counts, key=counts.get, reverse=True)[:top_n]


def _detect_governance_themes(text: str) -> list[str]:
    lower = text.lower()
    return [t for t in GOVERNANCE_THEMES if t in lower]


def enrich_metadata(
    chunk: Document,
    chunk_index: int,
    knowledge_set: str = "default",
) -> dict:
    text         = chunk.page_content
    lower        = text.lower()
    base         = chunk.metadata
    contains_pii = any(p in lower for p in PII_SIGNALS)

    return {
        "doc_id"            : str(uuid.uuid4()),
        "chunk_index"       : chunk_index,
        "knowledge_set"     : knowledge_set,
        "source_file"       : base.get("source_file", base.get("source", "unknown")),
        "file_type"         : base.get("file_type", "unknown"),
        "source_type"       : base.get("source_type", "file"),   # 'file' | 'databricks'
        "page"              : base.get("page"),
        "sheet"             : base.get("sheet"),
        # Databricks-specific passthrough
        "catalog"           : base.get("catalog"),
        "schema"            : base.get("schema"),
        "table"             : base.get("table"),
        "first_line"        : text.split("\n")[0][:120].strip() or "—",
        "char_count"        : len(text),
        "word_count"        : len(text.split()),
        "summary"           : text[:200].replace("\n", " "),
        "keywords"          : _extract_keywords(text),
        "governance_themes" : _detect_governance_themes(text),
        "department"        : _detect_department(text),
        "sensitivity"       : _detect_sensitivity(text),
        "contains_pii"      : contains_pii,
        "owner"             : "DAAS DMG",
        "domain"            : "Insurance",
        "ingested_at"       : datetime.now().isoformat(),
        "language"          : "en",
    }


def chunk_and_enrich(
    raw_documents: list[Document],
    knowledge_set: str = "default",
) -> list[Document]:
    """Split documents into enriched chunks and persist metadata to SQLite."""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=CHUNK_SIZE,
        chunk_overlap=CHUNK_OVERLAP,
        separators=["\n\n", "\n", ". ", " ", ""]
    )
    chunks   = splitter.split_documents(raw_documents)
    enriched = []
    records  = []

    for i, chunk in enumerate(chunks):
        metadata       = enrich_metadata(chunk, i, knowledge_set=knowledge_set)
        chunk.metadata = metadata
        enriched.append(chunk)
        records.append(metadata)

    # Batch insert to SQLite
    db_insert_batch("document_chunks", records)
    return enriched
