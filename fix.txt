kpi_number_rejected_row_person = 
VAR SelectedCategory =
    VALUES(AX_DQ_DIM_CATEGORY[dim_category_id]) -- captures the current filter from slicer

VAR GroupedErrorCounts =
    ADDCOLUMNS(
        SUMMARIZE(
            AX_DQ_FACT_LOG,
            AX_DQ_FACT_LOG[row_id],
            AX_DQ_DIM_PERSON[dim_person_id]
        ),
        "ErrorCount",
        CALCULATE(
            COUNTROWS(AX_DQ_FACT_LOG),
            AX_DQ_FACT_LOG[line_status] = "Error",
            AX_DQ_FACT_LOG[row_id] = EARLIER(AX_DQ_FACT_LOG[row_id]),
            -- Ensure the group also respects the category filter
           AX_DQ_FACT_LOG[functional_category_id] IN SelectedCategory
        )
    )

-- Find the maximum error count in the filtered context
VAR MaxErrorCount =
    MAXX(GroupedErrorCounts, [ErrorCount])

-- Filter groups with errors or maximum errors, respecting the category filter
VAR SelectedGroups =
    FILTER(
        GroupedErrorCounts,
        [ErrorCount] > 0 || [ErrorCount] = MaxErrorCount
    )

-- Count distinct row IDs with errors in these filtered groups
RETURN
    CALCULATE(
        DISTINCTCOUNT(AX_DQ_FACT_LOG[row_id]),
        AX_DQ_FACT_LOG[line_status] = "Error",
        TREATAS(
            SELECTCOLUMNS(
                SelectedGroups,
                "row_id", [row_id],
                "dim_person_id", [dim_person_id]
            ),
            AX_DQ_FACT_LOG[row_id],
            AX_DQ_DIM_PERSON[dim_person_id]
        ),
        -- Respect the category filter
        AX_DQ_FACT_LOG[functional_category_id] IN SelectedCategory
    )
kpi_distinct_rows = CALCULATE(DISTINCTCOUNT(AX_DQ_FACT_LOG[row_id]))
kpi_dq_score_without_weight = 

VAR AllControls =
    SUMX (
        VALUES ( AX_DQ_DIM_CATEGORY[dim_category_id] ),
        VAR currentCategoryId = AX_DQ_DIM_CATEGORY[dim_category_id]
        VAR currentTechCategory =
            LOOKUPVALUE (
                AX_DQ_DIM_CATEGORY[technical_category],
                AX_DQ_DIM_CATEGORY[dim_category_id], currentCategoryId
            )
        VAR RulesCount =
            CALCULATE (
                DISTINCTCOUNT ( AX_DQ_DIM_RULES[dim_rule_id] ),
                AX_DQ_DIM_RULES[category] = currentTechCategory
            )
        VAR RowCount =
            CALCULATE (
                DISTINCTCOUNT ( AX_DQ_FACT_LOG[row_id] ),
                AX_DQ_FACT_LOG[dim_category_id] = currentCategoryId
            )
        RETURN
            RulesCount * RowCount
    )

VAR TotalErrors =
    CALCULATE (
        COUNTROWS ( AX_DQ_FACT_LOG ),
        NOT ( ISBLANK ( AX_DQ_FACT_LOG[dim_rule_id] ) )
    )

RETURN
    IF (
        AllControls = 0,
        "NA",
        1 - DIVIDE ( TotalErrors, AllControls )
    )
kpi_dq_score_completeness = 
VAR Result =
CALCULATE(

    AX_DQ_DIM_RULES[dq_dimension] = "Completeness"
)
RETURN
IF(
    Result = "NA" || ISBLANK(Result),0, Result
)

kpi_number_accepted_row_person = 
VAR TotalRows = [kpi_distinct_rows]
VAR RejectedRows = [kpi_number_rejected_row_person]
VAR WarningRows = [kpi_number_warning_row_person]
VAR OkRows = [kpi_number_ok_row_person]
VAR SelectedStatuses = VALUES('AX_DQ_DIM_LINE_STATUS'[line_status])
VAR StatusCount = COUNTROWS(SelectedStatuses)
VAR ErrorSelected = CONTAINS(SelectedStatuses, 'AX_DQ_DIM_LINE_STATUS'[line_status], "Error")
VAR OkSelected = CONTAINS(SelectedStatuses, 'AX_DQ_DIM_LINE_STATUS'[line_status], "OK")
VAR WarningSelected = CONTAINS(SelectedStatuses, 'AX_DQ_DIM_LINE_STATUS'[line_status], "Warning")
 
RETURN
SWITCH(
    TRUE(),
    -- When all statuses are selected
    StatusCount = 3,
        -- Calculate based on your original logic
        SWITCH(
            TRUE(),
            ErrorSelected && OkSelected && WarningSelected,
                -- Error + Ok + Warning: Sum OkRows + WarningRows
                OkRows + WarningRows,
            ErrorSelected && OkSelected,
                -- Error + Ok: Sum OkRows + WarningRows
                OkRows + WarningRows,
            OkSelected && WarningSelected,
                -- Ok + Warning: Sum OkRows + WarningRows
                OkRows + WarningRows,
            -- Default fallback if other combinations
            TotalRows - RejectedRows
        ),
    -- When specific statuses are selected
    ErrorSelected && OkSelected,
        -- Error + Ok: Sum OkRows + WarningRows
        OkRows + WarningRows,
    ErrorSelected && WarningSelected,
        -- Error + Warning: Sum OkRows + WarningRows
        OkRows + WarningRows,
    OkSelected && WarningSelected,
        -- Ok + Warning: Sum OkRows + WarningRows
        OkRows + WarningRows,
    ErrorSelected,
        -- Only Error selected
        0,
    OkSelected,
        -- Only OK selected
        OkRows,
    WarningSelected,
        -- Only Warning selected
        WarningRows,
    -- Fallback
    TotalRows - RejectedRows
)

                FILTER(AX_DQ_FACT_LOG, AX_DQ_FACT_LOG[line_status] = "Error"),
                1

                AX_DQ_FACT_LOG,
                AX_DQ_FACT_LOG[row_id],
                AX_DQ_DIM_PERSON[dim_person_id]
