"""
services/memory.py

SQLite-backed conversation memory for GovernGPT.

Design:
- Each conversation is identified by (knowledge_set, session_id)
- session_id maps 1:1 to the frontend Knowledge Set project ID
- Messages stored in conversation_history table (auto-created on first use)
- History loaded fresh on every API call — no in-process state
- Compatible with LangChain >= 0.2.x (uses only langchain_core)

Table schema:
    conversation_history(
        id            INTEGER PRIMARY KEY AUTOINCREMENT,
        session_id    TEXT NOT NULL,
        knowledge_set TEXT NOT NULL,
        role          TEXT NOT NULL,   -- 'human' or 'ai'
        content       TEXT NOT NULL,
        run_id        TEXT,
        token_count   INTEGER,
        created_at    TEXT NOT NULL
    )
"""

from datetime import datetime
from langchain_core.messages import HumanMessage, AIMessage, BaseMessage
from db import db_query, _get_conn

# How many past Q&A turns to include as context window
# 1 turn = 1 human message + 1 AI message = 2 rows
DEFAULT_WINDOW_K = 6


# ── Table management ─────────────────────────────────────────────

def _ensure_history_table() -> None:
    """Create conversation_history table and index if not present."""
    with _get_conn() as conn:
        conn.execute("""
            CREATE TABLE IF NOT EXISTS conversation_history (
                id            INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id    TEXT NOT NULL,
                knowledge_set TEXT NOT NULL,
                role          TEXT NOT NULL,
                content       TEXT NOT NULL,
                run_id        TEXT,
                token_count   INTEGER DEFAULT 0,
                created_at    TEXT NOT NULL
            )
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_conv_session
            ON conversation_history (session_id, knowledge_set, id)
        """)
        conn.commit()


# ── Write ────────────────────────────────────────────────────────

def save_turn(
    session_id: str,
    knowledge_set: str,
    human_message: str,
    ai_message: str,
    run_id: str = "",
    token_count: int = 0,
) -> None:
    """
    Persist one complete Q&A turn (human + AI) to SQLite.
    Both messages are written in a single transaction.
    Called after every successful LLM response.
    """
    _ensure_history_table()
    now = datetime.now().isoformat()

    with _get_conn() as conn:
        conn.execute(
            """
            INSERT INTO conversation_history
                (session_id, knowledge_set, role, content, run_id, token_count, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?)
            """,
            (session_id, knowledge_set, "human", human_message, run_id, token_count, now),
        )
        conn.execute(
            """
            INSERT INTO conversation_history
                (session_id, knowledge_set, role, content, run_id, token_count, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?)
            """,
            (session_id, knowledge_set, "ai", ai_message, run_id, token_count, now),
        )
        conn.commit()


# ── Read ─────────────────────────────────────────────────────────

def load_history(
    session_id: str,
    knowledge_set: str,
    window_k: int = DEFAULT_WINDOW_K,
) -> list[dict]:
    """
    Load the last `window_k` Q&A turns from SQLite.
    Returns flat list of {role, content} dicts in chronological order (oldest first).
    window_k turns = window_k * 2 rows (each turn has 1 human + 1 AI row).
    """
    _ensure_history_table()
    rows = db_query(
        """
        SELECT role, content FROM (
            SELECT role, content, id
            FROM conversation_history
            WHERE session_id = ? AND knowledge_set = ?
            ORDER BY id DESC
            LIMIT ?
        )
        ORDER BY id ASC
        """,
        (session_id, knowledge_set, window_k * 2),
    )
    return rows


def load_history_as_messages(
    session_id: str,
    knowledge_set: str,
    window_k: int = DEFAULT_WINDOW_K,
) -> list[BaseMessage]:
    """
    Load history and return as LangChain message objects.
    Used directly by LCEL chains via MessagesPlaceholder.
    HumanMessage for role='human', AIMessage for role='ai'.
    """
    rows = load_history(session_id, knowledge_set, window_k)
    messages: list[BaseMessage] = []
    for row in rows:
        if row["role"] == "human":
            messages.append(HumanMessage(content=row["content"]))
        else:
            messages.append(AIMessage(content=row["content"]))
    return messages


def get_full_history(session_id: str, knowledge_set: str) -> list[dict]:
    """
    Load complete conversation history for a session with all metadata.
    Used by GET /api/query/history endpoint.
    """
    _ensure_history_table()
    return db_query(
        """
        SELECT id, session_id, knowledge_set, role, content,
               run_id, token_count, created_at
        FROM conversation_history
        WHERE session_id = ? AND knowledge_set = ?
        ORDER BY id ASC
        """,
        (session_id, knowledge_set),
    )


def get_session_stats(session_id: str, knowledge_set: str) -> dict:
    """Return summary statistics for a conversation session."""
    _ensure_history_table()
    rows = db_query(
        """
        SELECT
            COUNT(*)                                          AS total_messages,
            SUM(CASE WHEN role = 'human' THEN 1 ELSE 0 END)  AS human_turns,
            SUM(CASE WHEN role = 'ai'    THEN 1 ELSE 0 END)  AS ai_turns,
            MIN(created_at)                                   AS first_message_at,
            MAX(created_at)                                   AS last_message_at
        FROM conversation_history
        WHERE session_id = ? AND knowledge_set = ?
        """,
        (session_id, knowledge_set),
    )
    return rows[0] if rows else {}


# ── Delete ───────────────────────────────────────────────────────

def clear_history(session_id: str, knowledge_set: str) -> int:
    """
    Delete all conversation history for a session.
    Returns the number of rows deleted.
    Used by DELETE /api/query/history endpoint.
    """
    _ensure_history_table()
    with _get_conn() as conn:
        cursor = conn.execute(
            "DELETE FROM conversation_history WHERE session_id = ? AND knowledge_set = ?",
            (session_id, knowledge_set),
        )
        conn.commit()
        return cursor.rowcount
