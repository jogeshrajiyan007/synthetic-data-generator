def get_token(CLIENT_ID, CLIENT_SECRET):
    url = SECUREGPT_API_TOKEN_URL
    headersList = {"Content-Type": "application/x-www-form-urlencoded"}
    payload = f"grant_type=client_credentials&client_id={CLIENT_ID}&client_secret={CLIENT_SECRET}&scope=urn:grp:chatgpt"
    response = requests.request(method='POST', url=url, data=payload, headers=headersList)
    if response.status_code != 200:
        raise Exception(f'Request failed with status {response.status_code}, {response.text}')
    return response.json()["access_token"]

class SecureGPTChat(AzureChatOpenAI):
    def __init__(
        self,
        *,
        api_key: str | None = None,
        base_url: str | None = None,
        temperature: float | None = None,
        
    ) -> None:
        super().__init__(
            model= "gpt-4.1-2025-04-14", # "gpt-4o-2024-05-13",#"gpt-35-turbo-0301",
            api_key=api_key,
            api_version="2024-10-21",
            base_url = base_url,
            validate_base_url=False,
            http_client = httpx.Client(verify=False),
            default_headers={"Authorization": f"Bearer {api_key}"},
            temperature=0,
            top_p = 0.2,

        )
