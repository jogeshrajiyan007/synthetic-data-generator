import requests
import threading
import time
from config import CLIENT_ID, CLIENT_SECRET, TOKEN_URL


class TokenManager:
    """
    Thread-safe OAuth2 token manager.
    Automatically refreshes the token before expiry.
    """
    def __init__(self):
        self._token: str | None = None
        self._expires_at: float = 0
        self._lock = threading.Lock()

    def _fetch(self) -> dict:
        payload = (
            f"grant_type=client_credentials"
            f"&client_id={CLIENT_ID}"
            f"&client_secret={CLIENT_SECRET}"
            f"&scope=urn:grp:chatgpt"
        )
        resp = requests.post(
            TOKEN_URL,
            data=payload,
            headers={"Content-Type": "application/x-www-form-urlencoded"},
            verify=False
        )
        if resp.status_code != 200:
            raise RuntimeError(f"Token fetch failed [{resp.status_code}]: {resp.text}")
        return resp.json()

    def get_token(self) -> str:
        with self._lock:
            # Refresh if expired or within 60s of expiry
            if time.time() >= self._expires_at - 60:
                data = self._fetch()
                self._token = data["access_token"]
                self._expires_at = time.time() + data.get("expires_in", 3600)
            return self._token


# Singleton — import this everywhere
token_manager = TokenManager()
