"""
services/databricks/approved_sources.py

Per-session approved Databricks table registry.

Design:
- Each session (knowledge_set + session_id) has an explicit whitelist
  of Databricks tables the LLM is permitted to query via SQL.
- Tables are stored in the `approved_sources` SQLite table.
- The frontend sends the approved list with every query request
  (as the user's active source selection).
- The SQL agent validates every generated query against this list
  before execution — no table outside the list is ever queried.

Table schema:
    approved_sources(
        id            INTEGER PRIMARY KEY AUTOINCREMENT,
        session_id    TEXT NOT NULL,
        knowledge_set TEXT NOT NULL,
        connection_id TEXT NOT NULL,
        full_table    TEXT NOT NULL,   -- catalog.schema.table
        catalog       TEXT NOT NULL,
        schema        TEXT NOT NULL,
        table_name    TEXT NOT NULL,
        added_at      TEXT NOT NULL
    )
"""

from datetime import datetime
from db import _get_conn, db_query


# ── Table management ─────────────────────────────────────────────

def _ensure_approved_sources_table() -> None:
    with _get_conn() as conn:
        conn.execute("""
            CREATE TABLE IF NOT EXISTS approved_sources (
                id            INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id    TEXT NOT NULL,
                knowledge_set TEXT NOT NULL,
                connection_id TEXT NOT NULL,
                full_table    TEXT NOT NULL,
                catalog       TEXT NOT NULL,
                db_schema     TEXT NOT NULL,
                table_name    TEXT NOT NULL,
                added_at      TEXT NOT NULL,
                UNIQUE(session_id, knowledge_set, full_table)
            )
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_approved_session
            ON approved_sources (session_id, knowledge_set)
        """)
        conn.commit()


# ── Write ────────────────────────────────────────────────────────

def add_approved_source(
    session_id: str,
    knowledge_set: str,
    connection_id: str,
    catalog: str,
    schema: str,
    table_name: str,
) -> None:
    """Add a table to the approved list for this session. Idempotent."""
    _ensure_approved_sources_table()
    full_table = f"{catalog}.{schema}.{table_name}"
    now = datetime.now().isoformat()
    with _get_conn() as conn:
        conn.execute(
            """
            INSERT OR IGNORE INTO approved_sources
                (session_id, knowledge_set, connection_id, full_table,
                 catalog, db_schema, table_name, added_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (session_id, knowledge_set, connection_id,
             full_table, catalog, schema, table_name, now),
        )
        conn.commit()


def remove_approved_source(
    session_id: str,
    knowledge_set: str,
    full_table: str,
) -> int:
    """Remove a table from the approved list. Returns rows deleted."""
    _ensure_approved_sources_table()
    with _get_conn() as conn:
        cursor = conn.execute(
            "DELETE FROM approved_sources WHERE session_id=? AND knowledge_set=? AND full_table=?",
            (session_id, knowledge_set, full_table),
        )
        conn.commit()
        return cursor.rowcount


def clear_approved_sources(session_id: str, knowledge_set: str) -> int:
    """Remove all approved sources for a session."""
    _ensure_approved_sources_table()
    with _get_conn() as conn:
        cursor = conn.execute(
            "DELETE FROM approved_sources WHERE session_id=? AND knowledge_set=?",
            (session_id, knowledge_set),
        )
        conn.commit()
        return cursor.rowcount


# ── Read ─────────────────────────────────────────────────────────

def get_approved_sources(session_id: str, knowledge_set: str) -> list[dict]:
    """Return all approved tables for a session with full metadata."""
    _ensure_approved_sources_table()
    return db_query(
        """
        SELECT id, session_id, knowledge_set, connection_id,
               full_table, catalog, db_schema, table_name, added_at
        FROM approved_sources
        WHERE session_id = ? AND knowledge_set = ?
        ORDER BY added_at ASC
        """,
        (session_id, knowledge_set),
    )


def get_approved_table_names(session_id: str, knowledge_set: str) -> list[str]:
    """Return only the full table name strings (catalog.schema.table)."""
    rows = get_approved_sources(session_id, knowledge_set)
    return [r["full_table"] for r in rows]


def is_table_approved(
    session_id: str,
    knowledge_set: str,
    full_table: str,
) -> bool:
    """Check whether a specific table is in the approved list."""
    approved = get_approved_table_names(session_id, knowledge_set)
    return full_table.lower() in [t.lower() for t in approved]


def get_connection_for_session(session_id: str, knowledge_set: str) -> str | None:
    """
    Return the connection_id associated with the session's approved sources.
    All approved sources for a session should share the same connection.
    Returns None if no approved sources exist.
    """
    _ensure_approved_sources_table()
    rows = db_query(
        "SELECT connection_id FROM approved_sources "
        "WHERE session_id=? AND knowledge_set=? LIMIT 1",
        (session_id, knowledge_set),
    )
    return rows[0]["connection_id"] if rows else None


def sync_approved_sources(
    session_id: str,
    knowledge_set: str,
    connection_id: str,
    tables: list[str],
) -> dict:
    """
    Sync the approved list to exactly the provided list of full table names.
    Adds new entries, removes entries no longer in the list.
    Returns a summary of changes.
    """
    _ensure_approved_sources_table()
    current = set(get_approved_table_names(session_id, knowledge_set))
    desired = set(tables)

    to_add    = desired - current
    to_remove = current - desired

    for full_table in to_remove:
        remove_approved_source(session_id, knowledge_set, full_table)

    for full_table in to_add:
        parts = full_table.split(".")
        if len(parts) != 3:
            continue   # skip malformed entries
        cat, sch, tbl = parts
        add_approved_source(session_id, knowledge_set, connection_id, cat, sch, tbl)

    return {
        "added"  : sorted(to_add),
        "removed": sorted(to_remove),
        "total"  : len(desired),
    }
