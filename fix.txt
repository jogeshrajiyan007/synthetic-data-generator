// ── Chat ─────────────────────────────────────────────────────────
export interface ChatMessage {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: string
  // RAG metadata
  sources_used?: string[]
  chunks_retrieved?: number
  duration_seconds?: number
  run_id?: string
  rephrased_question?: string
  prompt_tokens?: number
  completion_tokens?: number
  total_tokens?: number
  mode?: string   // 'conversational_rag' | 'direct' | 'sql_agent' | 'hybrid'
  // SQL Agent metadata
  sql_intent?: string             // SIMPLE_LOOKUP | AGGREGATION | etc.
  sql_executed?: string           // the actual SQL that ran
  sql_row_count?: number
  sql_results?: Record<string, unknown>[]   // raw result rows for table display
  sql_think_aloud?: string        // agent's pre-execution narration
  governance_note?: string        // PII / DQ / ownership overlay
}

export interface QueryRequest {
  knowledge_set: string
  session_id: string
  question: string
  use_rag: boolean
  k: number
  window_k?: number
  enable_sql_agent?: boolean
  approved_tables?: string[]
  connection_id?: string
}

export interface QueryResponse {
  run_id: string
  session_id: string
  knowledge_set: string
  question: string
  rephrased_question: string
  answer: string
  mode: string
  sources_used: string[]
  chunks_retrieved: number
  model: string
  prompt_tokens: number
  completion_tokens: number
  total_tokens: number
  created_at: string
  duration_seconds: number
  // SQL Agent extensions
  sql_intent?: string | null
  sql_executed?: string | null
  sql_row_count?: number | null
  sql_results?: Record<string, unknown>[] | null
  sql_think_aloud?: string | null
  governance_note?: string | null
}
