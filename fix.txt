"""
routers/databricks.py

All Databricks-related API endpoints.

Endpoint groups:
  Connection management  — connect, list, delete, test
  Catalog browser        — catalogs, schemas, tables, table detail
  Profiling              — full statistical profile
  Ingestion              — profile + chunk + vectorise into knowledge set
"""

import time
from fastapi import APIRouter, HTTPException, Query, BackgroundTasks
from pydantic import BaseModel, Field
from typing import Optional

from services.databricks.client import (
    DatabricksClient, save_connection, get_connection,
    list_connections, delete_connection
)
from services.databricks.catalog import (
    list_catalogs, list_schemas, list_tables, get_table_detail
)
from services.databricks.profiler import profile_table
from services.databricks.ingestion import ingest_databricks_table
from db import db_query, db_insert

router = APIRouter(prefix="/api/databricks", tags=["Databricks"])


# ── Request / Response models ────────────────────────────────────

class ConnectRequest(BaseModel):
    connection_name: str = Field(..., min_length=1, max_length=120,
        description="Friendly name e.g. 'AXA Prod Databricks'")
    workspace_url: str = Field(..., description="e.g. https://adb-xxxx.azuredatabricks.net")
    pat: str = Field(..., min_length=10, description="Databricks Personal Access Token")
    http_path: str = Field(..., description="e.g. /sql/1.0/warehouses/xxxx")
    catalog: str = Field(default="hive_metastore",
        description="Default catalog (use 'hive_metastore' for legacy)")
    description: str = Field(default="")
    knowledge_set: str = Field(default="",
        description="Optional: associate connection with a knowledge set")


class ConnectResponse(BaseModel):
    connection_id: str
    connection_name: str
    workspace_url: str
    catalog: str
    status: str
    current_user: Optional[str] = None
    message: str


class ConnectionInfo(BaseModel):
    connection_id: str
    connection_name: str
    workspace_url: str
    http_path: str
    catalog: str
    description: str
    knowledge_set: str
    status: str
    last_tested_at: Optional[str]
    created_at: str


class IngestTableRequest(BaseModel):
    knowledge_set: str = Field(..., min_length=1, max_length=60)
    catalog: str       = Field(..., description="Catalog name")
    schema: str        = Field(..., description="Schema / database name")
    table: str         = Field(..., description="Table or view name")
    sample_rows: int   = Field(default=500_000, ge=1000, le=5_000_000,
        description="Max rows to profile for statistics")
    rebuild: bool      = Field(default=False,
        description="Drop existing collection before ingesting")


class IngestMultipleRequest(BaseModel):
    knowledge_set: str
    tables: list[dict] = Field(..., description="List of {catalog, schema, table} dicts")
    sample_rows: int   = Field(default=500_000, ge=1000, le=5_000_000)
    rebuild_first: bool = Field(default=False,
        description="Drop collection before ingesting first table")


# ── Connection management ────────────────────────────────────────

@router.post("/connect", response_model=ConnectResponse)
async def connect(request: ConnectRequest):
    """
    Test a Databricks connection, then save it with encrypted PAT.
    Returns the connection_id for all subsequent API calls.
    """
    # Test connectivity before saving
    test_client = None
    try:
        # Temporarily build a client without saving to test first
        from databricks import sql as db_sql
        import base64, requests as req

        ws = request.workspace_url.rstrip("/")
        if not ws.startswith("https://"):
            ws = f"https://{ws}"

        resp = req.get(
            f"{ws}/api/2.0/clusters/list",
            headers={"Authorization": f"Bearer {request.pat}"},
            timeout=10,
            verify=True,
        )
        if resp.status_code == 401:
            raise HTTPException(status_code=401, detail="Invalid PAT — authentication failed.")
        if resp.status_code not in (200, 403):
            # 403 is fine — means auth works but no cluster access
            raise HTTPException(
                status_code=400,
                detail=f"Workspace unreachable: {resp.status_code} {resp.text[:200]}"
            )
        current_user = resp.json().get("clusters", [{}])[0].get("creator_user_name", "unknown") \
            if resp.status_code == 200 else "authenticated"

    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(
            status_code=400,
            detail=f"Connection test failed: {str(e)}"
        )

    # Save encrypted connection
    connection_id = save_connection(
        connection_name=request.connection_name,
        workspace_url=request.workspace_url,
        pat=request.pat,
        http_path=request.http_path,
        catalog=request.catalog,
        description=request.description,
        knowledge_set=request.knowledge_set,
    )

    return ConnectResponse(
        connection_id  =connection_id,
        connection_name=request.connection_name,
        workspace_url  =request.workspace_url.rstrip("/"),
        catalog        =request.catalog,
        status         ="connected",
        current_user   =current_user,
        message        =f"Connection saved. PAT encrypted and stored securely.",
    )


@router.get("/connections", response_model=list[ConnectionInfo])
async def get_connections(
    knowledge_set: Optional[str] = Query(default=None)
):
    """List all active Databricks connections."""
    rows = list_connections(knowledge_set)
    return [ConnectionInfo(**r) for r in rows]


@router.delete("/connections/{connection_id}")
async def remove_connection(connection_id: str):
    """Soft-delete a Databricks connection."""
    deleted = delete_connection(connection_id)
    if not deleted:
        raise HTTPException(status_code=404, detail="Connection not found.")
    return {"status": "deleted", "connection_id": connection_id}


@router.post("/connections/{connection_id}/test")
async def test_connection(connection_id: str):
    """Re-test an existing saved connection."""
    try:
        with DatabricksClient(connection_id) as client:
            result = client.test_connection()
        if not result["success"]:
            raise HTTPException(status_code=400, detail=result.get("error"))
        return result
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))

# ── Ingestion ────────────────────────────────────────────────────

@router.post("/{connection_id}/ingest")
async def ingest_table(
    connection_id: str,
    request: IngestTableRequest,
):
    """
    Profile a table and ingest it into a GovernGPT knowledge set.
    The table schema, statistics, and governance metadata become
    searchable via the RAG engine.
    """
    try:
        with DatabricksClient(connection_id) as client:
            result = ingest_databricks_table(
                client=client,
                catalog=request.catalog,
                schema=request.schema,
                table=request.table,
                knowledge_set=request.knowledge_set,
                sample_rows=request.sample_rows,
                rebuild=request.rebuild,
            )
        return result
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Ingestion failed: {str(e)}")


@router.post("/{connection_id}/ingest-multiple")
async def ingest_multiple_tables(
    connection_id: str,
    request: IngestMultipleRequest,
    background_tasks: BackgroundTasks,
):
    """
    Ingest multiple tables from a Databricks schema into a knowledge set.
    Tables are processed sequentially in the background.
    Returns immediately with a job_id — poll /ingest-status/{job_id}.
    """
    import uuid as _uuid
    job_id = str(_uuid.uuid4())

    # Log job start
    db_insert("databricks_batch_jobs", {
        "job_id"        : job_id,
        "connection_id" : connection_id,
        "knowledge_set" : request.knowledge_set,
        "table_count"   : len(request.tables),
        "status"        : "queued",
        "created_at"    : __import__("datetime").datetime.now().isoformat(),
    })

    background_tasks.add_task(
        _run_batch_ingest,
        job_id=job_id,
        connection_id=connection_id,
        request=request,
    )

    return {
        "job_id"       : job_id,
        "status"       : "queued",
        "table_count"  : len(request.tables),
        "knowledge_set": request.knowledge_set,
        "message"      : f"Ingesting {len(request.tables)} table(s) in background. Poll /ingest-status/{job_id}",
    }


@router.get("/ingest-status/{job_id}")
async def get_ingest_status(job_id: str):
    """Poll the status of a background batch ingestion job."""
    rows = db_query(
        "SELECT * FROM databricks_batch_jobs WHERE job_id = ?",
        (job_id,),
    )
    if not rows:
        raise HTTPException(status_code=404, detail="Job not found.")
    return rows[0]


@router.get("/{connection_id}/ingestion-log")
async def get_ingestion_log(
    connection_id: str,
    knowledge_set: Optional[str] = Query(default=None),
    limit: int = Query(default=50, ge=1, le=200),
):
    """Get the ingestion history for a connection."""
    if knowledge_set:
        rows = db_query(
            "SELECT * FROM databricks_ingestion_log "
            "WHERE connection_id=? AND knowledge_set=? "
            "ORDER BY rowid DESC LIMIT ?",
            (connection_id, knowledge_set, limit),
        )
    else:
        rows = db_query(
            "SELECT * FROM databricks_ingestion_log "
            "WHERE connection_id=? ORDER BY rowid DESC LIMIT ?",
            (connection_id, limit),
        )
    return {"records": rows, "count": len(rows)}


# ── Background task ──────────────────────────────────────────────

async def _run_batch_ingest(
    job_id: str,
    connection_id: str,
    request: IngestMultipleRequest,
) -> None:
    """Background task for multi-table ingestion."""
    from datetime import datetime
    results   = []
    failed    = []

    try:
        with DatabricksClient(connection_id) as client:
            for i, tbl in enumerate(request.tables):
                try:
                    rebuild = request.rebuild_first and i == 0
                    result  = ingest_databricks_table(
                        client=client,
                        catalog=tbl["catalog"],
                        schema=tbl["schema"],
                        table=tbl["table"],
                        knowledge_set=request.knowledge_set,
                        sample_rows=request.sample_rows,
                        rebuild=rebuild,
                    )
                    results.append(result)
                except Exception as e:
                    failed.append({"table": tbl, "error": str(e)})

        import json as _json
        db_query(
            "UPDATE databricks_batch_jobs SET status=?, completed_at=?, results=? WHERE job_id=?",
        )
        # Use raw connection for UPDATE
        from db import _get_conn
        with _get_conn() as conn:
            conn.execute(
                "UPDATE databricks_batch_jobs "
                "SET status=?, completed_at=?, results=?, failed=? "
                "WHERE job_id=?",
                (
                    "completed" if not failed else "partial",
                    datetime.now().isoformat(),
                    _json.dumps(results),
                    _json.dumps(failed),
                    job_id,
                )
            )
            conn.commit()

    except Exception as e:
        from db import _get_conn
        from datetime import datetime
        with _get_conn() as conn:
            conn.execute(
                "UPDATE databricks_batch_jobs SET status='failed', error=?, completed_at=? WHERE job_id=?",
                (str(e), datetime.now().isoformat(), job_id)
            )
            conn.commit()
