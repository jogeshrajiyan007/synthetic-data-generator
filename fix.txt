import httpx
from langchain_openai import AzureChatOpenAI
from langchain_huggingface import HuggingFaceEmbeddings
from auth import token_manager
from config import CHAT_MODEL, API_BASE_URL, API_VERSION, MODELS_DIR


class SecureGPTChat(AzureChatOpenAI):
    def __init__(self, api_key: str, base_url: str):
        super().__init__(
            model=CHAT_MODEL,
            api_key=api_key,
            api_version=API_VERSION,
            base_url=base_url,
            validate_base_url=False,
            http_client=httpx.Client(verify=False),
            temperature=0,
            top_p=0.2,
        )


def get_llm() -> SecureGPTChat:
    """Returns a fresh LLM instance with a valid token."""
    token = token_manager.get_token()
    return SecureGPTChat(api_key=token, base_url=API_BASE_URL)


def get_embeddings() -> HuggingFaceEmbeddings:
    """Local embeddings — loads from disk, no network call."""
    return HuggingFaceEmbeddings(
        model_name=str(MODELS_DIR),
        model_kwargs={"device": "cpu"},
        encode_kwargs={"normalize_embeddings": True}
    )


# ── Module-level singletons (loaded once at startup) ───────────
_embeddings: HuggingFaceEmbeddings | None = None


def get_embeddings_singleton() -> HuggingFaceEmbeddings:
    global _embeddings
    if _embeddings is None:
        _embeddings = get_embeddings()
    return _embeddings
