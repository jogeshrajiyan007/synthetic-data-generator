"""
routers/document.py

Document generation endpoints for GovernGPT.

POST /api/document/generate   — Generate a .docx from the user's request
GET  /api/document/download/{filename} — Download a previously generated file
GET  /api/document/list        — List generated documents for a session
"""

from fastapi import APIRouter, HTTPException, Query
from fastapi.responses import FileResponse
from pydantic import BaseModel, Field
from pathlib import Path
from typing import Optional

from services.document_generator import generate_document, DOCS_DIR
from services.vectorstore import load_vector_store
from services.memory import load_history_as_messages
from langchain_core.output_parsers import StrOutputParser

router = APIRouter(prefix="/api/document", tags=["Document Generation"])


# ── Schemas ───────────────────────────────────────────────────────

class DocumentRequest(BaseModel):
    question     : str = Field(..., min_length=5, max_length=32000)
    knowledge_set: str = Field(..., min_length=1, max_length=60)
    session_id   : str = Field(..., min_length=1, max_length=120)
    window_k     : int = Field(default=6, ge=1, le=20)
    k            : int = Field(default=5, ge=1, le=20)


class DocumentResponse(BaseModel):
    success      : bool
    file_name    : str
    download_url : str
    title        : str
    section_count: int
    message      : str
    error        : Optional[str] = None


# ── Generate ──────────────────────────────────────────────────────

@router.post("/generate", response_model=DocumentResponse)
async def generate_doc(request: DocumentRequest):
    """
    Generate a professional .docx document from the user's request.

    Uses the knowledge set's RAG context plus conversation history
    to produce a fully structured Word document with AXA branding.
    """
    # Load RAG context from the knowledge set
    rag_context = ""
    try:
        from services.rag import _build_retriever, _format_docs
        vectordb     = load_vector_store(request.knowledge_set)
        retriever_fn = _build_retriever(vectordb, request.k)
        source_docs  = retriever_fn(request.question)
        rag_context  = _format_docs(source_docs)
    except Exception:
        pass  # Generate without RAG context if knowledge set unavailable

    # Load conversation history for follow-up doc requests
    chat_history = []
    try:
        chat_history = load_history_as_messages(
            request.session_id, request.knowledge_set, request.window_k
        )
    except Exception:
        pass

    # Generate the document
    result = generate_document(
        question      = request.question,
        rag_context   = rag_context,
        chat_history  = chat_history,
        knowledge_set = request.knowledge_set,
    )

    if not result["success"]:
        raise HTTPException(
            status_code=500,
            detail=f"Document generation failed: {result.get('error', 'Unknown error')}"
        )

    return DocumentResponse(
        success      = True,
        file_name    = result["file_name"],
        download_url = f"/api/document/download/{result['file_name']}",
        title        = result["title"],
        section_count= result["section_count"],
        message      = f"Successfully generated '{result['title']}' with {result['section_count']} sections.",
    )


# ── Download ──────────────────────────────────────────────────────

@router.get("/download/{filename}")
async def download_doc(filename: str):
    """
    Download a previously generated .docx file.
    Files are stored in the backend's generated_docs/ directory.
    """
    # Security: prevent path traversal
    if ".." in filename or "/" in filename or "\\" in filename:
        raise HTTPException(status_code=400, detail="Invalid filename.")

    file_path = DOCS_DIR / filename
    if not file_path.exists():
        raise HTTPException(status_code=404, detail="Document not found or expired.")

    return FileResponse(
        path        = str(file_path),
        filename    = filename,
        media_type  = "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    )


# ── List ──────────────────────────────────────────────────────────

@router.get("/list")
async def list_docs():
    """List all generated documents currently on disk."""
    files = sorted(DOCS_DIR.glob("*.docx"), key=lambda f: f.stat().st_mtime, reverse=True)
    return {
        "count": len(files),
        "documents": [
            {
                "file_name"   : f.name,
                "download_url": f"/api/document/download/{f.name}",
                "size_kb"     : round(f.stat().st_size / 1024, 1),
                "created_at"  : f.stat().st_mtime,
            }
            for f in files[:20]  # latest 20
        ]
    }
