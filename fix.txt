import pandas as pd
import numpy as np
import json
import plotly.express as px
import plotly.graph_objects as go

# Convert timestamps
df['created_at'] = pd.to_datetime(df['created_at'], errors='coerce')
df = df.dropna(subset=['created_at'])

# Time features
df['date'] = df['created_at'].dt.date
df['hour'] = df['created_at'].dt.hour
df['day'] = df['created_at'].dt.day_name()

# Fix sources_used (string → list)
def parse_sources(x):
    try:
        return json.loads(x) if isinstance(x, str) else x
    except:
        return []

df['sources_used'] = df['sources_used'].apply(parse_sources)

# Derived metrics
df['efficiency'] = df['completion_tokens'] / df['prompt_tokens']
df['answer_length'] = df['answer'].str.len()

print("📊 LLM OBSERVABILITY DASHBOARD")
print("="*50)

print(f"Total Queries        : {len(df)}")
print(f"Total Tokens         : {df['total_tokens'].sum():,.0f}")
print(f"Avg Tokens / Query   : {df['total_tokens'].mean():.2f}")
print(f"Avg Latency (sec)    : {df['duration_seconds'].mean():.2f}")
print(f"Avg Chunks Retrieved : {df['chunks_retrieved'].mean():.2f}")

fig = px.line(
    df.sort_values('created_at'),
    x="created_at",
    y="total_tokens",
    title="📈 Token Usage Over Time"
)
fig.show()

heatmap = df.pivot_table(
    values='total_tokens',
    index='hour',
    columns='date',
    aggfunc='sum'
)

fig = px.imshow(
    heatmap,
    aspect="auto",
    title="🔥 Token Usage Heatmap (Hour vs Date)"
)
fig.show()

fig = px.histogram(

fig = px.bar(
    sources.value_counts().reset_index(),

)

threshold = df['total_tokens'].mean() + 2 * df['total_tokens'].std()
df['anomaly'] = df['total_tokens'] > threshold

fig = px.scatter(

)
