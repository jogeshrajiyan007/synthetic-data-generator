import urllib3
from contextlib import asynccontextmanager
from datetime import datetime
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from routers.all_routers import (
    sources_router, logs_router,
    profile_router, reset_router,
    ks_router
)
from routers.ingest import router as ingest_router
from routers.query  import router as query_router
from services.vectorstore import load_vector_store, get_vector_count
from services.memory import _ensure_history_table
from db import db_count, db_table_exists
from models.schemas import HealthResponse

# Suppress corporate SSL warnings
urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)


# ── Startup / shutdown ──────────────────────────────────────────
@asynccontextmanager
async def lifespan(app: FastAPI):
    print("🏛️  Data Governance RAG API starting...")

    # Pre-load embeddings model (avoids cold start on first query)
    from llm import get_embeddings_singleton
    get_embeddings_singleton()
    print("✅ Embeddings model loaded.")

    # Ensure conversation_history table exists in SQLite
    _ensure_history_table()
    print("✅ Conversation history table ready.")

    yield
    print("👋 Shutting down.")


# ── App ──────────────────────────────────────────────────────────
app = FastAPI(
    title="Data Governance RAG API",
    description=(
        "AI-powered Data Governance framework for the insurance industry. "
        "Reads unstructured documents and generates governance intelligence "
        "using RAG over AXA SecureGPT (GPT-4o). "
        "Supports persistent multi-turn conversational memory via SQLite."
    ),
    version="2.0.0",
    lifespan=lifespan,
    docs_url="/docs",
    redoc_url="/redoc"
)

# ── CORS ─────────────────────────────────────────────────────────
app.add_middleware(
    CORSMiddleware,
    allow_origins=[
        "http://localhost:3000",
        "http://localhost:5173",
        "http://127.0.0.1:3000",
        "http://127.0.0.1:5173",
    ],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# ── Routers ──────────────────────────────────────────────────────
app.include_router(ingest_router)
app.include_router(query_router)
app.include_router(sources_router)
app.include_router(logs_router)
app.include_router(profile_router)
app.include_router(reset_router)
app.include_router(ks_router)


# ── Health check ─────────────────────────────────────────────────
@app.get("/health", response_model=HealthResponse, tags=["Health"])
async def health():
    chroma_count  = 0
    sources_count = 0
    try:
        vdb = load_vector_store()
        chroma_count = get_vector_count(vdb)
        if db_table_exists("document_chunks"):
            from db import db_query
            sources_count = len(db_query(
                "SELECT DISTINCT source_file FROM document_chunks"
            ))
    except Exception:
        pass

    return HealthResponse(
        status="healthy",
        version="2.0.0",
        chroma_vectors=chroma_count,
        sqlite_chunks=db_count("document_chunks"),
        sqlite_responses=db_count("llm_responses"),
        sqlite_conversation_messages=db_count("conversation_history"),
        sources_loaded=sources_count,
        timestamp=datetime.now().isoformat()
    )
