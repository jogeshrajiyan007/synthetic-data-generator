"""
services/databricks/catalog.py

Hive Metastore browser via Databricks SQL.

Supports:
- Unity Catalog (3-level: catalog.schema.table)
- Legacy Hive Metastore (2-level: schema.table under hive_metastore)
- Tables AND Views
- Full column schema including STRUCT / ARRAY / MAP nested types
"""

from __future__ import annotations
from services.databricks.client import DatabricksClient


# ── Catalog / Schema / Table listing ────────────────────────────

def list_catalogs(client: DatabricksClient) -> list[dict]:
    """
    List all catalogs the user has access to.
    Falls back to ['hive_metastore'] if Unity Catalog is not enabled.
    """
    try:
        rows = client.execute_query("SHOW CATALOGS")
        return [{"catalog": r.get("catalog", r.get("databaseName", ""))} for r in rows]
    except Exception:
        # Unity Catalog not enabled — return legacy metastore
        return [{"catalog": "hive_metastore"}]


def list_schemas(client: DatabricksClient, catalog: str = "hive_metastore") -> list[dict]:
    """List all schemas (databases) in a catalog."""
    try:
        if catalog == "hive_metastore":
            rows = client.execute_query("SHOW DATABASES")
            key  = "databaseName" if rows and "databaseName" in rows[0] else list(rows[0].keys())[0]
            return [{"schema": r[key], "catalog": catalog} for r in rows]
        else:
            rows = client.execute_query(f"SHOW SCHEMAS IN `{catalog}`")
            key  = "databaseName" if rows and "databaseName" in rows[0] else list(rows[0].keys())[0]
            return [{"schema": r[key], "catalog": catalog} for r in rows]
    except Exception as e:
        raise RuntimeError(f"Failed to list schemas in {catalog}: {e}") from e


def list_tables(
    client: DatabricksClient,
    catalog: str,
    schema: str,
    include_views: bool = True,
) -> list[dict]:
    """
    List all tables (and optionally views) in a schema.
    Returns table name, type, and comment.
    """
    try:
        if catalog == "hive_metastore":
            rows = client.execute_query(f"SHOW TABLES IN `{schema}`")
        else:
            rows = client.execute_query(f"SHOW TABLES IN `{catalog}`.`{schema}`")

        tables = []
        for r in rows:
            name      = r.get("tableName", r.get("name", ""))
            is_temp   = r.get("isTemporary", False)
            tbl_type  = "VIEW" if r.get("isView", False) else "TABLE"

            if is_temp:
                continue
            if not include_views and tbl_type == "VIEW":
                continue

            tables.append({
                "table_name"  : name,
                "table_type"  : tbl_type,
                "catalog"     : catalog,
                "schema"      : schema,
                "full_name"   : _fqn(catalog, schema, name),
            })
        return tables
    except Exception as e:
        raise RuntimeError(f"Failed to list tables in {catalog}.{schema}: {e}") from e


def get_table_detail(
    client: DatabricksClient,
    catalog: str,
    schema: str,
    table: str,
) -> dict:
    """
    Get full table detail: columns, types, comment, row count estimate,
    partition info, storage format, and location.
    Uses DESCRIBE TABLE EXTENDED which works on both Hive and Unity Catalog.
    """
    fqn = _fqn(catalog, schema, table)
    try:
        rows = client.execute_query(f"DESCRIBE TABLE EXTENDED {fqn}")
    except Exception as e:
        raise RuntimeError(f"Failed to describe {fqn}: {e}") from e

    columns     = []
    table_meta  = {}
    in_meta     = False

    for row in rows:
        col_name = str(row.get("col_name", "") or "").strip()
        data_type = str(row.get("data_type", "") or "").strip()
        comment   = str(row.get("comment", "") or "").strip()

        # Empty separator row signals start of table metadata section
        if col_name == "":
            in_meta = True
            continue

        if in_meta:
            # Key-value metadata pairs
            if col_name and data_type:
                table_meta[col_name.lower().replace(" ", "_")] = data_type
        else:
            if col_name.startswith("#"):
                continue
            columns.append({
                "column_name": col_name,
                "data_type"  : data_type,
                "comment"    : comment,
                "is_complex" : _is_complex_type(data_type),
                "parsed_type": _parse_complex_type(data_type),
            })

    return {
        "catalog"       : catalog,
        "schema"        : schema,
        "table"         : table,
        "full_name"     : fqn,
        "columns"       : columns,
        "column_count"  : len(columns),
        "table_type"    : table_meta.get("type", "MANAGED"),
        "provider"      : table_meta.get("provider", "delta"),
        "location"      : table_meta.get("location", ""),
        "comment"       : table_meta.get("comment", ""),
        "partition_cols": _extract_partition_cols(table_meta),
        "table_metadata": table_meta,
    }


# ── Complex type parser ──────────────────────────────────────────

def _is_complex_type(dtype: str) -> bool:
    d = dtype.lower().strip()
    return any(d.startswith(t) for t in ("struct<", "array<", "map<"))


def _parse_complex_type(dtype: str) -> dict:
    """
    Recursively parse complex Spark/Hive type definitions.

    Examples:
      struct<id:int,name:string>
      array<struct<id:int,value:double>>
      map<string,struct<key:string,val:int>>
    """
    dtype = dtype.strip()
    lower = dtype.lower()

    if lower.startswith("struct<"):
        inner  = dtype[7:-1]  # strip 'struct<' and '>'
        fields = _split_top_level(inner)
        result = {}
        for field in fields:
            if ":" in field:
                fname, ftype = field.split(":", 1)
                result[fname.strip()] = _parse_complex_type(ftype.strip())
        return {"type": "struct", "fields": result}

    elif lower.startswith("array<"):
        inner = dtype[6:-1]
        return {"type": "array", "element": _parse_complex_type(inner)}

    elif lower.startswith("map<"):
        inner  = dtype[4:-1]
        parts  = _split_top_level(inner)
        key_t  = _parse_complex_type(parts[0].strip()) if parts else "unknown"
        val_t  = _parse_complex_type(parts[1].strip()) if len(parts) > 1 else "unknown"
        return {"type": "map", "key": key_t, "value": val_t}

    else:
        return {"type": dtype}


def _split_top_level(s: str) -> list[str]:
    """
    Split a comma-separated type string at the top level only,
    ignoring commas inside nested angle brackets.
    e.g. 'id:int,name:struct<a:int,b:string>' → ['id:int', 'name:struct<a:int,b:string>']
    """
    parts  = []
    depth  = 0
    current = ""
    for ch in s:
        if ch == "<":
            depth += 1
            current += ch
        elif ch == ">":
            depth -= 1
            current += ch
        elif ch == "," and depth == 0:
            parts.append(current.strip())
            current = ""
        else:
            current += ch
    if current.strip():
        parts.append(current.strip())
    return parts


def _fqn(catalog: str, schema: str, table: str) -> str:
    """Build a fully qualified table name safe for SQL."""
    if catalog == "hive_metastore":
        return f"`{schema}`.`{table}`"
    return f"`{catalog}`.`{schema}`.`{table}`"


def _extract_partition_cols(meta: dict) -> list[str]:
    raw = meta.get("partition_columns", meta.get("partitioncolumns", ""))
    if not raw or raw in ("[]", ""):
        return []
    return [c.strip().strip("[]'\"") for c in raw.split(",") if c.strip()]


# ── Flat column list for profiling ───────────────────────────────

def flatten_schema(columns: list[dict], prefix: str = "") -> list[dict]:
    """
    Recursively flatten complex types to a list of leaf columns with their
    dot-notation paths. Used by the profiler to know what to sample.

    Returns list of:
    {
        "path"       : "address.city",       # dot-notation SQL path
        "data_type"  : "string",
        "is_leaf"    : True,
        "parent_path": "address",
        "depth"      : 1,
    }
    """
    flat = []
    for col in columns:
        name  = col["column_name"]
        dtype = col["data_type"]
        path  = f"{prefix}.{name}" if prefix else name

        if col["is_complex"]:
            parsed = col["parsed_type"]
            if parsed.get("type") == "struct":
                # Recurse into struct fields
                sub_cols = [
                    {
                        "column_name": fname,
                        "data_type"  : (ftype["type"] if isinstance(ftype, dict) else ftype),
                        "comment"    : "",
                        "is_complex" : isinstance(ftype, dict) and ftype.get("type") in ("struct", "array", "map"),
                        "parsed_type": ftype if isinstance(ftype, dict) else {"type": ftype},
                    }
                    for fname, ftype in parsed.get("fields", {}).items()
                ]
                flat.extend(flatten_schema(sub_cols, prefix=path))

            elif parsed.get("type") == "array":
                # Mark array as a special leaf — profiler samples element count
                flat.append({
                    "path"       : path,
                    "data_type"  : dtype,
                    "is_leaf"    : False,
                    "is_array"   : True,
                    "parent_path": prefix,
                    "depth"      : path.count("."),
                })

            else:
                # map or unknown complex — treat as opaque leaf
                flat.append({
                    "path"       : path,
                    "data_type"  : dtype,
                    "is_leaf"    : False,
                    "is_array"   : False,
                    "parent_path": prefix,
                    "depth"      : path.count("."),
                })
        else:
            flat.append({
                "path"       : path,
                "data_type"  : dtype,
                "is_leaf"    : True,
                "is_array"   : False,
                "parent_path": prefix,
                "depth"      : path.count("."),
            })

    return flat
