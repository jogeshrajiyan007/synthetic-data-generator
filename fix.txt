-- Refresh Script : would take 11 minutes for 3 moving months
SET SERVEROUTPUT ON;
DECLARE
  v_start_time  TIMESTAMP;
  v_end_time    TIMESTAMP;
BEGIN
    v_start_time := SYSTIMESTAMP;
    DBMS_OUTPUT.PUT_LINE('Refresh started at ' || v_start_time);
    DBMS_MVIEW.REFRESH
        (
        list            => 'AX_DQ_LOGS.AX_DQ_FACT_DETAILS_MTVW',
        method          => 'C', -- 'C' = Complete
        atomic_refresh  => FALSE, -- Ensures the refresh is all-or-nothing (safer)
        parallelism     => 16     -- Parallelism (test values like 2, 4, 8 depending on system load)
        );
    v_end_time := SYSTIMESTAMP;
    DBMS_OUTPUT.PUT_LINE('Refresh ended at ' || v_end_time);
END;
/
CREATE UNIQUE INDEX IDX_FD_ROW_NUMBER ON AX_DQ_LOGS.AX_DQ_FACT_DETAILS_MTVW (ROW_NUMBER);
