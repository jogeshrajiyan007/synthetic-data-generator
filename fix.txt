// ── Knowledge Set / Session ─────────────────────────────────────
export interface KnowledgeSet {
  id: string              // used as knowledge_set param in API calls
  project_name: string
  summary: string
  created_at: string
  file_count: number
  vector_count: number
  collection_name: string
}

export interface CreateKnowledgeSetPayload {
  project_name: string
  summary: string
}

// ── Files / Sources ─────────────────────────────────────────────
export interface SourceFile {
  id: string              // local uuid for UI key
  name: string
  size: number
  type: string
  extension: string
  status: 'uploading' | 'processing' | 'ready' | 'error'
  progress: number
  error?: string
  ingested_at?: string
  chunk_count?: number
  department?: string
  sensitivity?: string
  pii_chunks?: number
}

export interface IngestResponse {
  status: string
  knowledge_set: string
  collection_name: string
  files_processed: number
  files_failed: number
  total_chunks: number
  total_vectors: number
  file_details: FileDetail[]
  duration_seconds: number
  ingested_at: string
}

export interface FileDetail {
  file: string
  type: string
  status: string
  docs: number
  error?: string
}

// ── Chat ─────────────────────────────────────────────────────────
export interface ChatMessage {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: string
  sources_used?: string[]
  chunks_retrieved?: number
  duration_seconds?: number
  run_id?: string
}

export interface QueryRequest {
  knowledge_set: string
  question: string
  use_rag: boolean
  k: number
}

export interface QueryResponse {
  run_id: string
  knowledge_set: string
  question: string
  answer: string
  mode: string
  sources_used: string[]
  chunks_retrieved: number
  model: string
  created_at: string
  duration_seconds: number
}

// ── Stats ────────────────────────────────────────────────────────
export interface GovernanceStats {
  knowledge_set: string
  total_chunks: number
  total_llm_responses: number
  pii_chunks: number
  pii_percentage: number
  sensitivity_distribution: { sensitivity: string; count: number }[]
  department_distribution: { department: string; count: number }[]
}

export interface KnowledgeSetInfo {
  knowledge_set: string
  collection_name: string
  vector_count: number
}

export interface KnowledgeSetListResponse {
  total: number
  knowledge_sets: KnowledgeSetInfo[]
}
