fn_GetFactDetails = (StartDate as text, EndDate as text) =>
let
  pageSize = 10000,
  baseUrl = "https://ofa-acm-prd.axa.com/ords/",
  schemaPath = "ax_dq_logs/ax_dq/fact_details",
  // Function to fetch each page with date filters passed to API
  GetPage = (pgOffset as number) =>
        let
            response = Json.Document(
                Web.Contents(baseUrl, [
                    RelativePath = schemaPath,
                    Query = [
                        limit       = Text.From(pageSize),
                        offset      = Text.From(pgOffset),
                        start_date  = StartDate,
                        end_date    = EndDate
                    ]
                ])
            ),
            items = response[items]
        in
            items,
  // Paginate through all pages
  AllPages = List.Generate(
        ()    => [pgOffset = 0, page = GetPage(0)],
        each  List.Count([page]) > 0,
        each  [page = GetPage([pgOffset] + pageSize), pgOffset = [pgOffset] + pageSize],
        each  [page]
    ),
  Combined = List.Combine(AllPages),
  // ✅ Handle empty response — define all expected columns explicitly
  EmptyTable = #table(
        {
            "row_number", "data_point", "value", "row_id",
            "dim_date_id", "dim_person_id", "dim_category_id",
            "dim_entity_id", "line_status_code", "ax_filename",
            "ax_filename_date", "dim_rule_id", "error_message"
        },
        {}
    ),
  // ✅ If API returns empty, use EmptyTable to preserve schema
  Final = if List.IsEmpty(Combined)
            then EmptyTable
            else Table.FromRecords(Combined),
  #"Type modifié" = Table.TransformColumnTypes(Final, {{"dim_person_id", type text}, {"row_id", type text}}),
  #"Fractionner la colonne par délimiteur" = Table.SplitColumn(#"Type modifié", "dim_date_id", Splitter.SplitTextByDelimiter("T", QuoteStyle.Csv), {"dim_date_id.1", "dim_date_id.2"}),
  #"Type modifié1" = Table.TransformColumnTypes(#"Fractionner la colonne par délimiteur", {{"dim_date_id.1", type date}, {"dim_date_id.2", type time}}),
  #"Colonnes renommées" = Table.RenameColumns(#"Type modifié1", {{"dim_date_id.1", "dim_date_id"}}),
  #"Colonnes supprimées" = Table.RemoveColumns(#"Colonnes renommées", {"dim_date_id.2"}),
  #"Type modifié2" = Table.TransformColumnTypes(#"Colonnes supprimées", {{"dim_person_id", type text}, {"row_id", type text}}),
  #"Replaced Value" = Table.ReplaceValue(#"Type modifié2", null, "", Replacer.ReplaceValue, {"value"}),
  #"Changed Type" = Table.TransformColumnTypes(#"Replaced Value", {{"ax_filename_date", type text}}),
  Custom1 = Table.SelectRows(#"Changed Type", each true),
  #"Transform columns" = Table.TransformColumnTypes(Custom1, {{"row_number", type text}, {"data_point", type text}, {"value", type text}, {"dim_category_id", type text}, {"dim_entity_id", type text}, {"line_status_code", type text}, {"ax_filename", type text}, {"dim_rule_id", type text}, {"error_message", type text}}),
  #"Replace errors" = Table.ReplaceErrorValues(#"Transform columns", {{"row_number", null}, {"data_point", null}, {"value", null}, {"dim_category_id", null}, {"dim_entity_id", null}, {"line_status_code", null}, {"ax_filename", null}, {"dim_rule_id", null}, {"error_message", null}}),
  #"Duplicated column" = Table.DuplicateColumn(#"Replace errors", "dim_date_id", "dim_date_id - Copy"),
  #"Renamed columns" = Table.RenameColumns(#"Duplicated column", {{"dim_date_id - Copy", "reported_timestamp"}}),
  #"Changed column type" = Table.TransformColumnTypes(#"Renamed columns", {{"reported_timestamp", type datetime}})
in
  #"Changed column type"

Month_list = let
  StartDate = #date(2026,1,1),
  EndDate = #date(2026,2,28),

  MonthList = List.Generate(
  ()=> StartDate,
  each _ <= EndDate,
  each Date.AddMonths(_,1)
  ),

  TableMonths = Table.FromList(
  MonthList, Splitter.SplitByNothing(),{"Start"}
  ),

  AddEndDate = Table.AddColumn(TableMonths, "End", each Date.EndOfMonth([Start])),

  ConvertText = Table.TransformColumns(
  AddEndDate,
  {
    {"Start", each Date.ToText(_, "MM/dd/yyyy")},
    {"End", each Date.ToText(_, "MM/dd/yyyy")}
  }
  ),
  #"Invoked custom function" = Table.AddColumn(ConvertText, "Invoked custom function", each fn_GetFactDetails([Start], [End]))

in 
  #"Invoked custom function"

AX_DQ_FACT_DETAILS_HISTORICAL = let
  Source = Month_list,
  #"Expanded Invoked custom function" = Table.ExpandTableColumn(Source, "Invoked custom function", {"row_number", "data_point", "value", "row_id", "dim_date_id", "dim_person_id", "dim_category_id", "dim_entity_id", "line_status_code", "ax_filename", "ax_filename_date", "dim_rule_id", "error_message", "reported_timestamp"}, {"row_number", "data_point", "value", "row_id", "dim_date_id", "dim_person_id", "dim_category_id", "dim_entity_id", "line_status_code", "ax_filename", "ax_filename_date", "dim_rule_id", "error_message", "reported_timestamp"}),
  #"Removed columns" = Table.RemoveColumns(#"Expanded Invoked custom function", {"Start", "End"}),
  #"Transform columns" = Table.TransformColumnTypes(#"Removed columns", {{"row_number", type text}, {"data_point", type text}, {"value", type text}, {"row_id", type text}, {"dim_date_id", type text}, {"dim_person_id", type text}, {"dim_category_id", type text}, {"dim_entity_id", type text}, {"line_status_code", type text}, {"ax_filename", type text}, {"ax_filename_date", type text}, {"dim_rule_id", type text}, {"error_message", type text}, {"reported_timestamp", type text}}),
  #"Replace errors" = Table.ReplaceErrorValues(#"Transform columns", {{"row_number", null}, {"data_point", null}, {"value", null}, {"row_id", null}, {"dim_date_id", null}, {"dim_person_id", null}, {"dim_category_id", null}, {"dim_entity_id", null}, {"line_status_code", null}, {"ax_filename", null}, {"ax_filename_date", null}, {"dim_rule_id", null}, {"error_message", null}, {"reported_timestamp", null}})
in
  #"Replace errors"
