from pydantic import BaseModel, Field
from typing import Any, Optional
from datetime import datetime


# ── Ingest ──────────────────────────────────────────────────────

class IngestRequest(BaseModel):
    knowledge_set: str = Field(
        ..., min_length=1, max_length=60,
        description="Unique name for this knowledge set e.g. 'hr_analytics', 'claims_2024'"
    )
    rebuild: bool = Field(
        default=False,
        description="If True, drops the existing collection and rebuilds from scratch"
    )


class IngestResponse(BaseModel):
    status: str
    knowledge_set: str
    collection_name: str
    files_processed: int
    files_failed: int
    total_chunks: int
    total_vectors: int
    file_details: list[dict]
    duration_seconds: float
    ingested_at: str


# ── Query ───────────────────────────────────────────────────────

class QueryRequest(BaseModel):
    knowledge_set: str = Field(
        ..., min_length=1, max_length=60,
        description="Which knowledge set to query against"
    )
    session_id: str = Field(
        ..., min_length=1, max_length=120,
        description=(
            "Unique conversation session ID. Use the Knowledge Set project ID "
            "from the frontend. All messages with the same session_id share "
            "conversation memory."
        )
    )
    question: str = Field(..., min_length=3, max_length=32000)
    use_rag: bool = True
    k: int = Field(default=5, ge=1, le=20)
    window_k: int = Field(
        default=6, ge=1, le=20,
        description="Number of past Q&A turns to include as memory context"
    )
    # ── SQL Agent fields ─────────────────────────────────────────
    enable_sql_agent: bool = Field(
        default=True,
        description=(
            "If True, the engine first detects whether the question requires SQL, "
            "and if so runs the agentic SQL loop against approved Databricks tables. "
            "Falls back to RAG if no SQL intent is detected."
        )
    )
    approved_tables: list[str] = Field(
        default_factory=list,
        description=(
            "List of fully qualified table names (catalog.schema.table) the user has "
            "selected as active sources for this session. These are synced to the "
            "approved_sources SQLite table before query execution."
        )
    )
    connection_id: Optional[str] = Field(
        default=None,
        description="Databricks connection_id to use for SQL execution"
    )


class QueryResponse(BaseModel):
    run_id: str
    session_id: str
    knowledge_set: str
    question: str
    rephrased_question: str
    answer: str
    mode: str                          # 'conversational_rag' | 'direct' | 'sql_agent' | 'hybrid'
    sources_used: list[str]
    chunks_retrieved: int
    model: str
    prompt_tokens: int
    completion_tokens: int
    total_tokens: int
    created_at: str
    duration_seconds: float
    # ── SQL Agent extensions ─────────────────────────────────────
    sql_intent: Optional[str] = None   # SIMPLE_LOOKUP / AGGREGATION / etc.
    sql_executed: Optional[str] = None
    sql_row_count: Optional[int] = None
    sql_results: Optional[list[dict]] = None
    sql_think_aloud: Optional[str] = None
    governance_note: Optional[str] = None


# ── SQL Agent standalone endpoint schemas ────────────────────────

class SqlAgentRequest(BaseModel):
    knowledge_set: str = Field(..., min_length=1, max_length=60)
    session_id: str    = Field(..., min_length=1, max_length=120)
    question: str      = Field(..., min_length=3, max_length=32000)
    connection_id: str = Field(..., description="Databricks connection_id")
    approved_tables: list[str] = Field(
        ..., description="List of catalog.schema.table strings the user has approved"
    )
    window_k: int = Field(default=6, ge=1, le=20)


class SqlAgentResponse(BaseModel):
    run_id: str
    session_id: str
    knowledge_set: str
    question: str
    intent: str
    think_aloud: str
    sql: str
    row_count: int
    results: list[dict]
    interpretation: str
    governance_note: str
    mode: str
    error: Optional[str] = None
    duration_seconds: float
    model: str
    created_at: str
    approved_tables: list[str]


# ── Approved Sources management schemas ──────────────────────────

class ApprovedSourceEntry(BaseModel):
    full_table   : str
    catalog      : str
    schema       : str
    table_name   : str
    connection_id: str
    added_at     : str


class ApprovedSourcesResponse(BaseModel):
    session_id   : str
    knowledge_set: str
    total        : int
    sources      : list[ApprovedSourceEntry]


class SyncApprovedSourcesRequest(BaseModel):
    session_id   : str = Field(..., min_length=1, max_length=120)
    knowledge_set: str = Field(..., min_length=1, max_length=60)
    connection_id: str = Field(..., description="Databricks connection_id")
    tables       : list[str] = Field(
        ..., description="Complete list of catalog.schema.table to approve"
    )


class SyncApprovedSourcesResponse(BaseModel):
    session_id   : str
    knowledge_set: str
    added        : list[str]
    removed      : list[str]
    total        : int


# ── Conversation History ─────────────────────────────────────────

class ConversationMessage(BaseModel):
    id: int
    session_id: str
    knowledge_set: str
    role: str                      # 'human' | 'ai'
    content: str
    run_id: Optional[str] = None
    token_count: Optional[int] = None
    created_at: str


class ConversationHistoryResponse(BaseModel):
    session_id: str
    knowledge_set: str
    total_messages: int
    turns: int                     # total_messages / 2
    messages: list[ConversationMessage]


class ConversationStatsResponse(BaseModel):
    session_id: str
    knowledge_set: str
    total_messages: int
    human_turns: int
    ai_turns: int
    first_message_at: Optional[str] = None
    last_message_at: Optional[str] = None


class ClearHistoryResponse(BaseModel):
    status: str
    session_id: str
    knowledge_set: str
    messages_deleted: int

class FileProfileResponse(BaseModel):

    total_nulls: int
