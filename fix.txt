import { SourcePanel } from '@/components/sources/SourcePanel'
import { ChatWindow } from '@/components/chat/ChatWindow'
import type { KnowledgeSet, SourceFile, ChatMessage } from '@/types'

interface WorkspaceProps {
  session: KnowledgeSet
  files: SourceFile[]
  messages: ChatMessage[]
  onBack: () => void
  onAddFiles: (files: SourceFile[]) => void
  onUpdateFile: (id: string, updates: Partial<SourceFile>) => void
  onRemoveFile: (id: string) => void
  onAddMessage: (msg: ChatMessage) => void

export function Workspace({

  onAddMessage, onUpdateMessage, onClearChat,

  return (
    <motion.div
      initial={{ opacity: 0 }}

          onUpdateFile={onUpdateFile}

        />
      </div>

      <div className="w-px bg-white/5 flex-shrink-0" />

      {/* Right panel — Chat (70-75%) */}
      <div className="flex-1 h-full overflow-hidden">
        <ChatWindow
          session={session}
          messages={messages}
          onAddMessage={onAddMessage}
          onUpdateMessage={onUpdateMessage}
          onClearChat={onClearChat}
        />
      </div>
    </motion.div>
  )
}
