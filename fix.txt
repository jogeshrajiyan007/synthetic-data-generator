def get_token():
    global client_id, client_secret, api_token_url
 
    url = api_token_url
    headersList = {"Content-Type": "application/x-www-form-urlencoded"}
    payload = f"grant_type=client_credentials&client_id={client_id}&client_secret={client_secret}&scope=urn:grp:chatgpt"
    response = requests.request(method='POST', url=url, data=payload, headers=headersList)
    if response.status_code != 200:
        raise Exception(f'Request failed with status {response.status_code}, {response.text}')
    token_info = response.json()
    access_token = token_info['access_token']
    return access_token

def query_llm(user_prompt, gpt4o_llm):
    try:
        prompt = ChatPromptTemplate.from_messages(
            [
                (
                    "system",
                    "You are a helpful assistant.",
                ),
                ("human", "{input}"),
            ]
        )
 
        chain = prompt | gpt4o_llm
        llm_response = chain.invoke(
            {
                "input": user_prompt,
            }
        )

        # Check if response is an object with a 'content' attribute (like AIMessage)
        # if hasattr(llm_response, 'content'):
        #     llm_response_text = llm_response.content
        # else:
        #     llm_response_text = llm_response  # assume it's a string

        # Parse the response as JSON
        response_json = {
            "content": llm_response.content,
            "model": llm_response.response_metadata.get("model_name"),
            "model_provider": llm_response.response_metadata.get("model_provider"),
            "finish_reason": llm_response.response_metadata.get("finish_reason"),
            "run_id": llm_response.id,
            "usage": {
                "prompt_tokens": llm_response.usage_metadata.get("input_tokens"),
                "completion_tokens": llm_response.usage_metadata.get("output_tokens"),
                "total_tokens": llm_response.usage_metadata.get("total_tokens"),
            },
            "content_filter_results": llm_response.response_metadata.get("content_filter_results"),
            "tool_calls": llm_response.tool_calls,
            "additional_kwargs": llm_response.additional_kwargs,
        }
        return response_json
        
    except json.JSONDecodeError:
        print("Response is not valid JSON:", llm_response)
        return {"error": "Invalid JSON response from the model."}
    except Exception as e:
        print(str(e))
        return f"Error while calling LLM: {str(e)}"
