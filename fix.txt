"""
services/databricks/ingestion.py

Converts a Databricks table profile into LangChain Documents
and feeds them through the existing chunking → vectorstore pipeline.

The profile document is structured to maximise RAG retrieval quality:
- Schema section (column names, types, comments)
- Statistics section (per-column numerical/categorical stats)
- Governance section (PII, sensitivity, regulatory signals)
- Sample data section (5 random rows for context)

Each table produces 1 primary profile Document + optional sub-Documents
for deeply nested STRUCTs (one per top-level struct column).
"""

from __future__ import annotations
import json
from datetime import datetime

from langchain_core.documents import Document

from services.databricks.client import DatabricksClient
from services.databricks.profiler import profile_table
from services.databricks.catalog import _fqn
from services.chunking import chunk_and_enrich
from services.vectorstore import build_vector_store, add_to_vector_store, load_vector_store
from db import db_insert


# ── Public entry point ───────────────────────────────────────────

def ingest_databricks_table(
    client: DatabricksClient,
    catalog: str,
    schema: str,
    table: str,
    knowledge_set: str,
    sample_rows: int = 500_000,
    rebuild: bool = False,
) -> dict:
    """
    Full pipeline: profile → document → chunk → embed → vectorstore.

    Returns a summary dict with ingestion stats.
    """
    start = datetime.now()

    # ── 1. Profile the table ──────────────────────────────────────
    profile = profile_table(client, catalog, schema, table, sample_rows)

    # ── 2. Convert profile to Documents ──────────────────────────
    documents = profile_to_documents(profile)

    # ── 3. Fetch sample rows ──────────────────────────────────────
    sample_doc = _fetch_sample_document(client, catalog, schema, table)
    if sample_doc:
        documents.append(sample_doc)

    # ── 4. Chunk and enrich with governance metadata ───────────────
    chunks = chunk_and_enrich(documents, knowledge_set=knowledge_set)

    # ── 5. Store vectors ──────────────────────────────────────────
    if rebuild:
        from services.vectorstore import delete_vector_store
        delete_vector_store(knowledge_set)

    try:
        vectordb = load_vector_store(knowledge_set)
        add_to_vector_store(vectordb, chunks)
    except Exception:
        vectordb = build_vector_store(chunks, knowledge_set=knowledge_set)

    # ── 6. Log ingestion to SQLite ────────────────────────────────
    duration = (datetime.now() - start).total_seconds()
    log = {
        "ingestion_type"  : "databricks",
        "knowledge_set"   : knowledge_set,
        "connection_id"   : client.connection_id,
        "source"          : f"{catalog}.{schema}.{table}",
        "catalog"         : catalog,
        "schema"          : schema,
        "table"           : table,
        "row_count"       : profile["row_count"],
        "column_count"    : profile["column_count"],
        "chunks_created"  : len(chunks),
        "documents"       : len(documents),
        "sampled"         : str(profile["sampled"]),
        "duration_seconds": str(round(duration, 2)),
        "ingested_at"     : datetime.now().isoformat(),
    }
    db_insert("databricks_ingestion_log", log)

    return {
        "status"         : "success",
        "source"         : f"{catalog}.{schema}.{table}",
        "knowledge_set"  : knowledge_set,
        "row_count"      : profile["row_count"],
        "column_count"   : profile["column_count"],
        "documents"      : len(documents),
        "chunks"         : len(chunks),
        "sampled"        : profile["sampled"],
        "duration_seconds": round(duration, 2),
        "profiled_at"    : profile["profiled_at"],
    }


# ── Profile → Documents ──────────────────────────────────────────

def profile_to_documents(profile: dict) -> list[Document]:
    """
    Convert a table profile dict into one or more LangChain Documents.

    Structure:
    - Doc 1: Schema + governance overview (always created)
    - Doc 2: Column statistics narrative (always created)
    - Doc N: One additional doc per top-level STRUCT column (if present)
    """
    docs = []

    fqn      = profile["full_name"]
    gov      = profile["governance"]
    cols     = profile["columns"]
    profiles = profile["column_profiles"]

    # ── Document 1: Schema + Overview ────────────────────────────
    schema_sections = [
        f"TABLE: {fqn}",
        f"SOURCE TYPE: Databricks {'Unity Catalog' if '.' in fqn.replace('`','') else 'Hive Metastore'}",
        f"TABLE TYPE: {profile['table_type']} | FORMAT: {profile['provider']}",
        f"ROWS: {profile['row_count']:,} | COLUMNS: {profile['column_count']}",
        f"SAMPLED: {'Yes (statistics based on sample)' if profile['sampled'] else 'No (full table)'}",
        f"PARTITIONED BY: {', '.join(profile['partition_cols']) or 'None'}",
        f"",
        f"GOVERNANCE CLASSIFICATION:",
        f"  Sensitivity    : {gov['sensitivity']}",
        f"  Department     : {gov['department']}",
        f"  PII Columns    : {gov['pii_column_count']} column(s) — {', '.join(gov['pii_columns']) or 'None detected'}",
        f"  Themes         : {', '.join(gov['governance_themes']) or 'None detected'}",
        f"  Recommended Owner: {gov['recommended_owner']}",
        f"",
        f"COLUMN SCHEMA ({profile['column_count']} columns):",
    ]

    for col in cols:
        pii_flag = " ⚠️ PII" if col["column_name"] in gov["pii_columns"] else ""
        complex_flag = " [COMPLEX]" if col["is_complex"] else ""
        schema_sections.append(
            f"  {col['column_name']} : {col['data_type']}{complex_flag}{pii_flag}"
            + (f" — {col['comment']}" if col.get('comment') else "")
        )

    docs.append(Document(
        page_content="\n".join(schema_sections),
        metadata={
            "source_file" : fqn,
            "source_type" : "databricks",
            "file_type"   : "databricks_schema",
            "catalog"     : profile["catalog"],
            "schema"      : profile["schema"],
            "table"       : profile["table"],
            "row_count"   : str(profile["row_count"]),
            "col_count"   : str(profile["column_count"]),
            "sensitivity" : gov["sensitivity"],
            "department"  : gov["department"],
            "has_pii"     : str(gov["pii_column_count"] > 0),
            "profiled_at" : profile["profiled_at"],
        }
    ))

    # ── Document 2: Column Statistics ────────────────────────────
    stats_sections = [
        f"COLUMN STATISTICS FOR: {fqn}",
        f"Profiled at: {profile['profiled_at']}",
        f"",
    ]

    for cp in profiles:
        col_name = cp["column_name"]
        dtype    = cp["data_type"]
        null_pct = cp.get("null_pct")
        col_type = cp.get("col_type", "unknown")
        stats    = cp.get("stats", {})
        gov_col  = cp.get("governance", {})

        section = [f"\nCOLUMN: {col_name} | TYPE: {dtype} | CATEGORY: {col_type}"]

        if null_pct is not None:
            section.append(f"  Null Rate: {null_pct}%  ({cp.get('null_count', 'N/A')} nulls)")
        if cp.get("unique_count") is not None:
            section.append(f"  Unique Values: {cp['unique_count']:,}")

        if col_type == "numerical" and stats:
            section.append(
                f"  Mean: {stats.get('mean_val')}  Median: {stats.get('median')}"
                f"  StdDev: {stats.get('stddev_val')}"
                f"  Min: {stats.get('min_val')}  Max: {stats.get('max_val')}"
            )
            if stats.get("skewness") is not None:
                section.append(f"  Skewness: {stats.get('skewness')}  Kurtosis: {stats.get('kurtosis')}")

        elif col_type == "categorical" and stats.get("value_counts"):
            vc = stats["value_counts"]
            vp = stats.get("value_percentages", {})
            vals = " | ".join(
                f"{k}: {v} ({vp.get(k, '?')}%)" for k, v in list(vc.items())[:10]
            )
            section.append(f"  Top Values: {vals}")

        elif col_type == "datetime" and stats:
            section.append(
                f"  Earliest: {stats.get('earliest')}  "
                f"Latest: {stats.get('latest')}  "
                f"Distinct Dates: {stats.get('distinct_dates')}"
            )

        elif col_type == "array" and stats:
            section.append(
                f"  Array Size — Min: {stats.get('array_size_min')}  "
                f"Max: {stats.get('array_size_max')}  "
                f"Avg: {stats.get('array_size_avg')}"
            )

        if gov_col.get("contains_pii"):
            section.append(
                f"  ⚠️  PII: {gov_col.get('pii_type', 'DETECTED')} — "
                f"Masking: {gov_col.get('masking_technique', 'required')}"
            )

        stats_sections.extend(section)

    # Data quality summary
    dup = profile.get("duplicate_estimate", {})
    stats_sections.extend([
        f"",
        f"DATA QUALITY SUMMARY:",
        f"  Duplicate Rows (estimate): "
        f"{dup.get('duplicate_count', 'N/A')} "
        f"({dup.get('duplicate_pct', 'N/A')}%)",
        f"  PII Column Count: {gov['pii_column_count']}",
        f"  Sensitivity: {gov['sensitivity']}",
    ])

    docs.append(Document(
        page_content="\n".join(stats_sections),
        metadata={
            "source_file" : fqn,
            "source_type" : "databricks",
            "file_type"   : "databricks_stats",
            "catalog"     : profile["catalog"],
            "schema"      : profile["schema"],
            "table"       : profile["table"],
            "sensitivity" : gov["sensitivity"],
            "profiled_at" : profile["profiled_at"],
        }
    ))

    # ── Extra docs for top-level STRUCT columns ───────────────────
    for col in cols:
        if col["is_complex"] and col["data_type"].lower().startswith("struct<"):
            struct_doc = _struct_column_document(fqn, col, profile)
            if struct_doc:
                docs.append(struct_doc)

    return docs


def _struct_column_document(fqn: str, col: dict, profile: dict) -> Document | None:
    """Create a dedicated document for a STRUCT column's nested schema."""
    parsed = col.get("parsed_type", {})
    if parsed.get("type") != "struct":
        return None

    lines = [
        f"NESTED STRUCT SCHEMA: {fqn}.{col['column_name']}",
        f"Parent Table: {fqn}",
        f"Column: {col['column_name']} ({col['data_type']})",
        f"",
        f"STRUCT FIELDS:",
    ]

    def _render_fields(fields: dict, indent: int = 2) -> None:
        prefix = "  " * indent
        for fname, ftype in fields.items():
            if isinstance(ftype, dict):
                t = ftype.get("type", "unknown")
                if t == "struct":
                    lines.append(f"{prefix}{fname}: STRUCT")
                    _render_fields(ftype.get("fields", {}), indent + 1)
                elif t == "array":
                    elem = ftype.get("element", {})
                    lines.append(f"{prefix}{fname}: ARRAY<{elem.get('type', '?')}>")
                else:
                    lines.append(f"{prefix}{fname}: {t}")
            else:
                lines.append(f"{prefix}{fname}: {ftype}")

    _render_fields(parsed.get("fields", {}))

    return Document(
        page_content="\n".join(lines),
        metadata={
            "source_file"     : fqn,
            "source_type"     : "databricks",
            "file_type"       : "databricks_struct",
            "struct_column"   : col["column_name"],
            "catalog"         : profile["catalog"],
            "schema"          : profile["schema"],
            "table"           : profile["table"],
            "profiled_at"     : profile["profiled_at"],
        }
    )


def _fetch_sample_document(
    client: DatabricksClient,
    catalog: str,
    schema: str,
    table: str,
    n: int = 5,
) -> Document | None:
    """Fetch a small random sample for contextual RAG grounding."""
    fqn = _fqn(catalog, schema, table)
    try:
        rows = client.execute_query(
            f"SELECT * FROM {fqn} ORDER BY RAND() LIMIT {n}"
        )
        if not rows:
            return None

        lines = [
            f"SAMPLE ROWS FROM: {fqn}",
            f"(Random sample of {len(rows)} rows for context)",
            f"",
        ]
        for i, row in enumerate(rows, 1):
            # Truncate long values for embedding efficiency
            clean_row = {
                k: (str(v)[:100] if v is not None else "NULL")
                for k, v in row.items()
            }
            lines.append(f"Row {i}: {json.dumps(clean_row, ensure_ascii=False)}")

        return Document(
            page_content="\n".join(lines),
            metadata={
                "source_file": fqn,
                "source_type": "databricks",
                "file_type"  : "databricks_sample",
                "catalog"    : catalog,
                "schema"     : schema,
                "table"      : table,
            }
        )
    except Exception:
        return None
