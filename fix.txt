import os
import socket
import pymssql
from contextlib import contextmanager

class AppLogger:
    def __init__(self):
        server = os.getenv('DB_SERVER')
        database = os.getenv('DB_NAME')
        user = os.getenv('DB_USER')
        password = os.getenv('DB_PASSWORD')
        self.conn_params = {
            'server': server,
            'user': user,
            'password': password,
            'database': database
        }

    def _validate_env_vars(self):
        required_vars = ['OWNER_NAME', 'APP_NAME']
        missing = [var for var in required_vars if not os.getenv(var)]
        if missing:
            raise ValueError(f"Missing env vars: {', '.join(missing)}")

    @contextmanager
    def get_db(self):
        conn = pymssql.connect(**self.conn_params)
        try:
            yield conn
            conn.commit()
        finally:
            conn.close()

    def log(self, 
            owner_name: str, 
            appname: str, 
            hostname: str, 
            model_name: str, 
            request_id: str, 
            request_created: int, 
            completion_tokens: int, 
            prompt_tokens: int, 
            total_tokens: int, 
            response_time: float) -> bool:
        try:
            with self.get_db() as conn:
                with conn.cursor() as cur:
                    cur.execute("""
                        INSERT INTO logger.app_log (
                            owner_name, appname, hostname, model_name,
                            request_id, request_created, completion_tokens,
                            prompt_tokens, total_tokens, response_time
                        ) VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
                    """, (
                        owner_name, appname, hostname, model_name,
                        request_id, request_created, completion_tokens,
                        prompt_tokens, total_tokens, response_time
                    ))
            return True
        except Exception as e:
            print(f"Error: {e}")
            return False
