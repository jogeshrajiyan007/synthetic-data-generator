"""
routers/sql_agent.py

Dedicated SQL Agent endpoints.

Provides:
  POST /api/sql-agent/query              — run the full 6-step agentic SQL loop
  GET  /api/sql-agent/approved-sources   — list approved tables for a session
  POST /api/sql-agent/approved-sources/sync — sync approved table list
  DELETE /api/sql-agent/approved-sources — clear approved table list
  GET  /api/sql-agent/log                — SQL execution history for a session
"""

from fastapi import APIRouter, HTTPException, Query as QParam

from services.databricks.sql_agent import run_sql_agent
from services.databricks.approved_sources import (
    get_approved_sources,
    sync_approved_sources,
    clear_approved_sources,
    get_connection_for_session,
)
from services.memory import load_history_as_messages
from db import db_query
from models.schemas import (
    SqlAgentRequest, SqlAgentResponse,
    ApprovedSourcesResponse, ApprovedSourceEntry,
    SyncApprovedSourcesRequest, SyncApprovedSourcesResponse,
)

router = APIRouter(prefix="/api/sql-agent", tags=["SQL Agent"])


# ── SQL Agent query ──────────────────────────────────────────────

@router.post("/query", response_model=SqlAgentResponse)
async def sql_agent_query(request: SqlAgentRequest):
    """
    Run the full 6-step agentic SQL loop:
    1. Intent detection
    2. Think aloud
    3. SQL generation
    4. Execute via Databricks SQL API (read-only enforced)
    5. Interpret results
    6. Governance overlay

    Only tables listed in `approved_tables` can be queried.
    Prohibited write operations are blocked at the validation layer.
    """
    # Sync approved sources to SQLite registry
    if request.approved_tables:
        sync_approved_sources(
            session_id    = request.session_id,
            knowledge_set = request.knowledge_set,
            connection_id = request.connection_id,
            tables        = request.approved_tables,
        )

    # Load conversation history for context
    chat_history = load_history_as_messages(
        request.session_id,
        request.knowledge_set,
        request.window_k,
    )

    try:
        result = run_sql_agent(
            question      = request.question,
            session_id    = request.session_id,
            knowledge_set = request.knowledge_set,
            chat_history  = chat_history,
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"SQL agent error: {str(e)}")

    return SqlAgentResponse(**result)


# ── Approved Sources management ──────────────────────────────────

@router.get("/approved-sources", response_model=ApprovedSourcesResponse)
async def get_approved(
    session_id:    str = QParam(...),
    knowledge_set: str = QParam(...),
):
    """List all approved Databricks tables for a session."""
    rows = get_approved_sources(session_id, knowledge_set)
    sources = [
        ApprovedSourceEntry(
            full_table   = r["full_table"],
            catalog      = r["catalog"],
            schema       = r["db_schema"],
            table_name   = r["table_name"],
            connection_id= r["connection_id"],
            added_at     = r["added_at"],
        )
        for r in rows
    ]
    return ApprovedSourcesResponse(
        session_id    = session_id,
        knowledge_set = knowledge_set,
        total         = len(sources),
        sources       = sources,
    )


@router.post("/approved-sources/sync", response_model=SyncApprovedSourcesResponse)
async def sync_approved(request: SyncApprovedSourcesRequest):
    """
    Sync the approved table list for a session.
    Replaces the current list with the provided one.
    Called by the frontend whenever the user changes their active source selection.
    """
    summary = sync_approved_sources(
        session_id    = request.session_id,
        knowledge_set = request.knowledge_set,
        connection_id = request.connection_id,
        tables        = request.tables,
    )
    return SyncApprovedSourcesResponse(
        session_id    = request.session_id,
        knowledge_set = request.knowledge_set,
        added         = summary["added"],
        removed       = summary["removed"],
        total         = summary["total"],
    )


@router.delete("/approved-sources")
async def clear_approved(
    session_id:    str = QParam(...),
    knowledge_set: str = QParam(...),
):
    """Remove all approved sources for a session."""
    deleted = clear_approved_sources(session_id, knowledge_set)
    return {
        "status"          : "cleared",
        "session_id"      : session_id,
        "knowledge_set"   : knowledge_set,
        "entries_deleted" : deleted,
    }


# ── SQL execution log ────────────────────────────────────────────

@router.get("/log")
async def get_sql_log(
    session_id:    str = QParam(...),
    knowledge_set: str = QParam(...),
    limit: int = QParam(default=50, ge=1, le=200),
):
    """
    Get the SQL execution history for a session.
    Shows intent, SQL executed, row count, duration, errors.
    """
    rows = db_query(
        """
        SELECT run_id, question, intent, sql, row_count,
               duration_seconds, has_error, created_at
        FROM sql_agent_log
        WHERE session_id = ? AND knowledge_set = ?
        ORDER BY rowid DESC LIMIT ?
        """,
        (session_id, knowledge_set, limit),
    )
    return {
        "session_id"   : session_id,
        "knowledge_set": knowledge_set,
        "count"        : len(rows),
        "records"      : rows,
    }
