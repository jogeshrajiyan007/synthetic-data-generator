import type {
  QueryResponse,
  ConversationHistoryResponse,
  ConversationStatsResponse,
  DatabricksConnection,
  DatabricksConnectResponse,
  DatabricksCatalog,
  DatabricksSchema,
  DatabricksTable,
  DatabricksTableDetail,
  DatabricksIngestResult,
  ApprovedSourcesResponse,
  DocumentGenerateRequest,
  DocumentGenerateResponse,
} from '@/types'

const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000'

async function request<T>(path: string, options?: RequestInit): Promise<T> {
  const res = await fetch(`${API_URL}${path}`, {
    headers: { 'Content-Type': 'application/json', ...options?.headers },
    ...options,
  })
  if (!res.ok) {
    const err = await res.json().catch(() => ({ detail: res.statusText }))
    throw new Error(err.detail || `Request failed: ${res.status}`)
  }
  return res.json()
}

export const api = {

  // ── Knowledge Sets ──────────────────────────────────────────────
  listKnowledgeSets: () =>
    request<{ total: number; knowledge_sets: { knowledge_set: string; collection_name: string; vector_count: number }[] }>('/api/knowledge-sets/'),

  deleteKnowledgeSet: (ks: string) =>
    request(`/api/knowledge-sets/${encodeURIComponent(ks)}`, { method: 'DELETE' }),

  // ── Ingest (files) ──────────────────────────────────────────────
  ingestLoadingZone: (knowledge_set: string, rebuild = false) =>
    request('/api/ingest/', {
      method: 'POST',
      body: JSON.stringify({ knowledge_set, rebuild }),
    }),

  uploadFiles: async (knowledge_set: string, files: File[]) => {
    const form = new FormData()
    form.append('knowledge_set', knowledge_set)
    files.forEach(f => form.append('files', f))
    const res = await fetch(`${API_URL}/api/ingest/upload`, {
      method: 'POST',
      body: form,
    })
    if (!res.ok) {
      const err = await res.json().catch(() => ({ detail: res.statusText }))
      throw new Error(err.detail || `Upload failed: ${res.status}`)
    }
    return res.json()
  },

  // ── Query (RAG + SQL Agent hybrid) ──────────────────────────────
  // The backend auto-detects SQL intent when enable_sql_agent=true
  // and approved_tables are present, routing to the SQL agent or
  // falling back to conversational RAG.
  query: (payload: {
    knowledge_set: string
    session_id: string
    question: string
    use_rag: boolean
    k: number
    window_k?: number
    enable_sql_agent?: boolean
    approved_tables?: string[]   // catalog.schema.table list
    connection_id?: string       // Databricks connection for SQL execution
    excluded_tables?: string[]   // tables removed from ingest — excluded from RAG
  }): Promise<QueryResponse> =>
    request('/api/query/', {
      method: 'POST',
      body: JSON.stringify({
        window_k        : 6,
        enable_sql_agent: true,
        approved_tables : [],
        ...payload,
      }),
    }),

  // ── Conversation History ─────────────────────────────────────────
  getConversationHistory: (session_id: string, knowledge_set: string): Promise<ConversationHistoryResponse> =>
    request(`/api/query/history?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}`),

  getConversationStats: (session_id: string, knowledge_set: string): Promise<ConversationStatsResponse> =>
    request(`/api/query/history/stats?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}`),

  clearConversationHistory: (session_id: string, knowledge_set: string) =>
    request(`/api/query/history?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}`, { method: 'DELETE' }),

  // ── Sources / Stats / Profile ───────────────────────────────────
  getSources: (knowledge_set: string) =>
    request(`/api/sources/?knowledge_set=${encodeURIComponent(knowledge_set)}`),

  getStats: (knowledge_set: string) =>
    request(`/api/logs/stats?knowledge_set=${encodeURIComponent(knowledge_set)}`),

  profileFile: (filename: string) =>
    request(`/api/profile/${encodeURIComponent(filename)}`),

  health: () => request('/health'),

  // ── SQL Agent ───────────────────────────────────────────────────
  sqlAgent: {
    /** Standalone SQL agent query (bypasses RAG entirely) */
    query: (payload: {
      knowledge_set: string
      session_id: string
      question: string
      connection_id: string
      approved_tables: string[]
      window_k?: number
    }) =>
      request('/api/sql-agent/query', {
        method: 'POST',
        body: JSON.stringify({ window_k: 6, ...payload }),
      }),

    /** List approved tables for a session */
    getApprovedSources: (session_id: string, knowledge_set: string): Promise<ApprovedSourcesResponse> =>
      request(`/api/sql-agent/approved-sources?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}`),

    /** Sync the approved table list (replaces current list) */
    syncApprovedSources: (payload: {
      session_id: string
      knowledge_set: string
      connection_id: string
      tables: string[]
    }) =>
      request('/api/sql-agent/approved-sources/sync', {
        method: 'POST',
        body: JSON.stringify(payload),
      }),

    /** Remove all approved sources for a session */
    clearApprovedSources: (session_id: string, knowledge_set: string) =>
      request(`/api/sql-agent/approved-sources?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}`, { method: 'DELETE' }),

    /** SQL execution history for a session */
    getLog: (session_id: string, knowledge_set: string, limit = 50) =>
      request(`/api/sql-agent/log?session_id=${encodeURIComponent(session_id)}&knowledge_set=${encodeURIComponent(knowledge_set)}&limit=${limit}`),
  },

  // ── Databricks: Connection Management ───────────────────────────
  databricks: {
    connect: (payload: {
      connection_name: string
      workspace_url: string
      pat: string
      http_path: string
      catalog?: string
      description?: string
      knowledge_set?: string
    }): Promise<DatabricksConnectResponse> =>
      request('/api/databricks/connect', { method: 'POST', body: JSON.stringify(payload) }),

    listConnections: (knowledge_set?: string): Promise<DatabricksConnection[]> =>
      request(`/api/databricks/connections${knowledge_set ? `?knowledge_set=${encodeURIComponent(knowledge_set)}` : ''}`),

    deleteConnection: (connection_id: string) =>
      request(`/api/databricks/connections/${encodeURIComponent(connection_id)}`, { method: 'DELETE' }),

    testConnection: (connection_id: string) =>
      request(`/api/databricks/connections/${encodeURIComponent(connection_id)}/test`, { method: 'POST' }),

    listCatalogs: (connection_id: string): Promise<{ catalogs: DatabricksCatalog[] }> =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/catalogs`),

    listSchemas: (connection_id: string, catalog: string): Promise<{ catalog: string; schemas: DatabricksSchema[] }> =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/schemas?catalog=${encodeURIComponent(catalog)}`),

    listTables: (connection_id: string, catalog: string, schema: string, include_views = true): Promise<{ tables: DatabricksTable[]; count: number }> =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/tables?catalog=${encodeURIComponent(catalog)}&schema=${encodeURIComponent(schema)}&include_views=${include_views}`),

    getTableDetail: (connection_id: string, catalog: string, schema: string, table: string): Promise<DatabricksTableDetail> =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/table-detail?catalog=${encodeURIComponent(catalog)}&schema=${encodeURIComponent(schema)}&table=${encodeURIComponent(table)}`),

    profileTable: (connection_id: string, catalog: string, schema: string, table: string, sample_rows = 500000) =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/profile?catalog=${encodeURIComponent(catalog)}&schema=${encodeURIComponent(schema)}&table=${encodeURIComponent(table)}&sample_rows=${sample_rows}`),

    ingestTable: (connection_id: string, payload: {
      knowledge_set: string; catalog: string; db_schema: string; table: string
      sample_rows?: number; rebuild?: boolean
    }): Promise<DatabricksIngestResult> =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/ingest`, {
        method: 'POST', body: JSON.stringify(payload),
      }),

    ingestMultiple: (connection_id: string, payload: {
      knowledge_set: string
      tables: { catalog: string; db_schema: string; table: string }[]
      sample_rows?: number; rebuild_first?: boolean
    }) =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/ingest-multiple`, {
        method: 'POST', body: JSON.stringify(payload),
      }),

    getIngestStatus: (job_id: string) =>
      request(`/api/databricks/ingest-status/${encodeURIComponent(job_id)}`),

    getIngestionLog: (connection_id: string, knowledge_set?: string) =>
      request(`/api/databricks/${encodeURIComponent(connection_id)}/ingestion-log${knowledge_set ? `?knowledge_set=${encodeURIComponent(knowledge_set)}` : ''}`),
  },

  // ── Document Generation ─────────────────────────────────────────
  document: {
    /** Generate a professional .docx from the user's request + RAG context */
    generate: (payload: DocumentGenerateRequest): Promise<DocumentGenerateResponse> =>
      request('/api/document/generate', {
        method: 'POST',
        body: JSON.stringify({ window_k: 6, k: 5, ...payload }),
      }),

    /** Download URL helper — returns full URL for the file */
    downloadUrl: (filename: string): string =>
      `${API_URL}/api/document/download/${encodeURIComponent(filename)}`,

    /** List previously generated documents */
    list: () =>
      request<{ count: number; documents: { file_name: string; download_url: string; size_kb: number; created_at: number }[] }>('/api/document/list'),
  },
}
