from pydantic import BaseModel, Field
from typing import Any, Optional
from datetime import datetime


# ── Ingest ──────────────────────────────────────────────────────

class IngestRequest(BaseModel):
    knowledge_set: str = Field(
        ..., min_length=1, max_length=60,
        description="Unique name for this knowledge set e.g. 'hr_analytics', 'claims_2024'"
    )
    rebuild: bool = Field(
        default=False,
        description="If True, drops the existing collection and rebuilds from scratch"
    )


class IngestResponse(BaseModel):
    status: str
    knowledge_set: str
    collection_name: str
    files_processed: int
    files_failed: int
    total_chunks: int
    total_vectors: int
    file_details: list[dict]
    duration_seconds: float
    ingested_at: str


# ── Query ───────────────────────────────────────────────────────

class QueryRequest(BaseModel):
    knowledge_set: str = Field(
        ..., min_length=1, max_length=60,
        description="Which knowledge set to query against"
    )
    session_id: str = Field(
        ..., min_length=1, max_length=120,
        description=(
            "Unique conversation session ID. Use the Knowledge Set project ID "
            "from the frontend. All messages with the same session_id share "
            "conversation memory."
        )
    )
    question: str = Field(..., min_length=3, max_length=32000)
    use_rag: bool = True
    k: int = Field(default=5, ge=1, le=20)
    window_k: int = Field(
        default=6, ge=1, le=20,
        description="Number of past Q&A turns to include as memory context"
    )


class QueryResponse(BaseModel):
    run_id: str
    session_id: str
    knowledge_set: str
    question: str
    rephrased_question: str        # ← standalone question after history condensing
    answer: str
    mode: str
    sources_used: list[str]
    chunks_retrieved: int
    model: str
    # ── Token tracking ─────────────────────────────────────────
    prompt_tokens: int
    completion_tokens: int
    total_tokens: int
    # ──────────────────────────────────────────────────────────
    created_at: str
    duration_seconds: float


# ── Conversation History ─────────────────────────────────────────

class ConversationMessage(BaseModel):
    id: int
    session_id: str
    knowledge_set: str
    role: str                      # 'human' | 'ai'
    content: str
    run_id: Optional[str] = None
    token_count: Optional[int] = None
    created_at: str


class ConversationHistoryResponse(BaseModel):
    session_id: str
    knowledge_set: str
    total_messages: int
    turns: int                     # total_messages / 2
    messages: list[ConversationMessage]


class ConversationStatsResponse(BaseModel):
    session_id: str
    knowledge_set: str
    total_messages: int
    human_turns: int
    ai_turns: int
    first_message_at: Optional[str] = None
    last_message_at: Optional[str] = None


class ClearHistoryResponse(BaseModel):
    status: str
    session_id: str
    knowledge_set: str
    messages_deleted: int


# ── Sources ─────────────────────────────────────────────────────

class SourceFile(BaseModel):
    source_file: str
    file_type: str
    chunk_count: int
    department: str
    sensitivity: str
    pii_chunks: int
    ingested_at: str


class SourcesResponse(BaseModel):
    knowledge_set: str
    total_sources: int
    total_chunks: int
    sources: list[SourceFile]


# ── Profile ─────────────────────────────────────────────────────

class ColumnProfile(BaseModel):
    column_name: str
    dtype: str
    null_count: int
    null_pct: float
    unique_count: int
    col_type: str
    stats: dict[str, Any]


class FileProfileResponse(BaseModel):
    source_file: str
    row_count: int
    col_count: int
    duplicate_rows: int
    total_nulls: int
    columns: list[ColumnProfile]
    strong_correlations: list[dict]
    governance: dict[str, Any]


# ── Logs ────────────────────────────────────────────────────────

class LogsResponse(BaseModel):
    table: str
    knowledge_set: Optional[str] = None
    total_records: int
    limit: int
    offset: int
    records: list[dict]


# ── Reset ───────────────────────────────────────────────────────

class ResetResponse(BaseModel):
    status: str
    message: str
    cleared: list[str]


# ── Health ──────────────────────────────────────────────────────

class HealthResponse(BaseModel):
    status: str
    version: str
    chroma_vectors: int
    sqlite_chunks: int
    sqlite_responses: int
    sqlite_conversation_messages: int   # ← new
    sources_loaded: int
    timestamp: str


# ── Knowledge Sets ───────────────────────────────────────────────

class KnowledgeSetInfo(BaseModel):
    knowledge_set: str
    collection_name: str
    vector_count: int


class KnowledgeSetListResponse(BaseModel):
    total: int
    knowledge_sets: list[KnowledgeSetInfo]


class KnowledgeSetDeleteResponse(BaseModel):
    status: str
    knowledge_set: str
    message: str
