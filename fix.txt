services/rag.py — Conversational RAG using LCEL (no deprecated chains)
Compatible with langchain >= 0.2.x
"""
import time
import uuid
from datetime import datetime

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.messages import HumanMessage, AIMessage
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough, RunnableLambda
from langchain_community.vectorstores import Chroma
from langchain_core.documents import Document

from llm import get_llm
from db import db_insert
from config import CHAT_MODEL, RETRIEVAL_K
from services.memory import save_turn, load_history, DEFAULT_WINDOW_K


# ── Prompts ──────────────────────────────────────────────────────

CONDENSE_PROMPT = ChatPromptTemplate.from_messages([
    ("system",
     "Given the conversation history and a follow-up question, "
     "rewrite the follow-up as a standalone question that captures all context. "
     "Return ONLY the rewritten question, nothing else."),
    MessagesPlaceholder(variable_name="chat_history"),
    ("human", "{question}"),
])

QA_PROMPT = ChatPromptTemplate.from_messages([
    ("system",
     """You are an expert Data Governance Advisor specialising in the insurance industry, \
with deep knowledge of GDPR, Solvency II, IFRS 17, and enterprise data management.

Retrieved document context:
{context}

Use the conversation history provided to:
- Refer back to findings from earlier in the conversation
- Build on previously identified entities, owners, issues or recommendations
- Avoid repeating analysis already done unless asked to expand
- Connect insights across prompts (e.g. "Based on the PII I identified earlier...")

Always cite which source file insights are drawn from.
If the answer cannot be found in context or history, say so clearly. Do not hallucinate."""),
    MessagesPlaceholder(variable_name="chat_history"),
    ("human", "{question}"),
])


# ── Helpers ──────────────────────────────────────────────────────

def _load_lc_history(session_id: str, knowledge_set: str, window_k: int) -> list:
    """Load SQLite history and convert to LangChain message objects."""
    rows = load_history(session_id, knowledge_set, window_k)
    messages = []
    for row in rows:
        if row["role"] == "human":
            messages.append(HumanMessage(content=row["content"]))
        else:
            messages.append(AIMessage(content=row["content"]))
    return messages


def _build_retriever(vectordb: Chroma, k: int):
    base = vectordb.as_retriever(
        search_type="mmr",
        search_kwargs={"k": k * 2, "fetch_k": k * 6},
    )

    def multi_source(query: str) -> list[Document]:
        docs = base.invoke(query)
        seen: dict[str, Document] = {}
        for doc in docs:
            src = doc.metadata.get("source_file", "unknown")
            if src not in seen:
                seen[src] = doc
        if len(seen) < 2:
            all_srcs = list({
                d.metadata.get("source_file", "unknown")
                for d in vectordb.similarity_search("data governance summary", k=50)
            })
            for src in all_srcs:
                if src not in seen:
                    results = vectordb.similarity_search(
                        query, k=2, filter={"source_file": src}
                    )
                    if results:
                        seen[src] = results[0]
        truncated = []
        for doc in seen.values():
            if len(doc.page_content) > 1500:
                doc = Document(
                    page_content=doc.page_content[:1500] + "\n[...truncated]",
                    metadata=doc.metadata,
                )
            truncated.append(doc)
        return truncated

    return multi_source


def _format_docs(docs: list[Document]) -> str:
    parts = []
    for doc in docs:
        src = doc.metadata.get("source_file", "unknown")
        parts.append(f"[Source: {src}]\n{doc.page_content}")
    return "\n\n---\n\n".join(parts)


# ── Main query function ──────────────────────────────────────────

def query_governance(
    question: str,
    vectordb: Chroma,
    knowledge_set: str = "default",
    session_id: str = "",
    use_rag: bool = True,
    k: int = RETRIEVAL_K,
    window_k: int = DEFAULT_WINDOW_K,
) -> dict:
    start  = time.time()
    run_id = str(uuid.uuid4())

    if not session_id:
        session_id = knowledge_set

    # Load history from SQLite as LangChain message objects
    chat_history = _load_lc_history(session_id, knowledge_set, window_k)

    llm = get_llm()

    if use_rag:
        retriever_fn = _build_retriever(vectordb, k)

        # Step 1 — Condense question using history
        if chat_history:
            condense_chain = CONDENSE_PROMPT | llm | StrOutputParser()
            rephrased = condense_chain.invoke({
                "chat_history": chat_history,
                "question": question,
            })
        else:
            rephrased = question

        # Step 2 — Retrieve chunks using rephrased question
        source_docs = retriever_fn(rephrased)
        context     = _format_docs(source_docs)
        sources     = list({
            doc.metadata.get("source_file", "unknown") for doc in source_docs
        })

        # Step 3 — Generate answer with context + history
        qa_chain = QA_PROMPT | llm | StrOutputParser()
        answer   = qa_chain.invoke({
            "context"     : context,
            "chat_history": chat_history,
            "question"    : question,
        })

        # Extract token usage
        prompt_tokens     = 0
        completion_tokens = 0
        total_tokens      = 0

        result = {
            "run_id"             : run_id,
            "session_id"         : session_id,
            "knowledge_set"      : knowledge_set,
            "question"           : question,
            "rephrased_question" : rephrased,
            "answer"             : answer,
            "mode"               : "conversational_rag",
            "sources_used"       : sources,
            "chunks_retrieved"   : len(source_docs),
            "model"              : CHAT_MODEL,
            "prompt_tokens"      : prompt_tokens,
            "completion_tokens"  : completion_tokens,
            "total_tokens"       : total_tokens,
            "created_at"         : datetime.now().isoformat(),
            "duration_seconds"   : round(time.time() - start, 2),
        }

    else:
        # Direct mode — no retrieval, still context-aware
        direct_prompt = ChatPromptTemplate.from_messages([
            ("system",
             "You are a Data Governance Expert for the insurance industry. "
             "Build on the conversation history when answering."),
            MessagesPlaceholder(variable_name="chat_history"),
            ("human", "{question}"),
        ])
        chain    = direct_prompt | llm | StrOutputParser()
        answer   = chain.invoke({"chat_history": chat_history, "question": question})

        result = {
            "run_id"             : run_id,
            "session_id"         : session_id,
            "knowledge_set"      : knowledge_set,
            "question"           : question,
            "rephrased_question" : question,
            "answer"             : answer,
            "mode"               : "direct",
            "sources_used"       : [],
            "chunks_retrieved"   : 0,
            "model"              : CHAT_MODEL,
            "prompt_tokens"      : 0,
            "completion_tokens"  : 0,
            "total_tokens"       : 0,
            "created_at"         : datetime.now().isoformat(),
            "duration_seconds"   : round(time.time() - start, 2),
        }

    # Persist to SQLite
    db_insert("llm_responses", result)
    save_turn(
        session_id    = session_id,
        knowledge_set = knowledge_set,
        human_message = question,
        ai_message    = answer,
        run_id        = run_id,
    )

    return result
