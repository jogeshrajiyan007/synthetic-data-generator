import json
from langchain_core.documents import Document
from langchain_community.vectorstores import Chroma
from llm import get_embeddings_singleton
from config import CHROMA_DIR

COLLECTION_NAME = "governance_docs"


def _flatten_metadata(chunk: Document) -> Document:
    """Chroma rejects list/bool — flatten all metadata values to strings."""
    flat = {
        k: (json.dumps(v) if isinstance(v, (list, bool, dict))
            else str(v) if v is not None else "")
        for k, v in chunk.metadata.items()
    }
    return Document(page_content=chunk.page_content, metadata=flat)


def build_vector_store(chunks: list[Document]) -> Chroma:
    safe_chunks = [_flatten_metadata(c) for c in chunks]
    embeddings  = get_embeddings_singleton()

    vectordb = Chroma.from_documents(
        documents=safe_chunks,
        embedding=embeddings,
        persist_directory=str(CHROMA_DIR),
        collection_name=COLLECTION_NAME,
    )
    vectordb.persist()
    return vectordb


def load_vector_store() -> Chroma:
    return Chroma(
        persist_directory=str(CHROMA_DIR),
        embedding_function=get_embeddings_singleton(),
        collection_name=COLLECTION_NAME,
    )


def add_to_vector_store(vectordb: Chroma, chunks: list[Document]) -> Chroma:
    """Add new chunks to an existing vector store."""
    safe_chunks = [_flatten_metadata(c) for c in chunks]
    vectordb.add_documents(safe_chunks)
    vectordb.persist()
    return vectordb


def get_vector_count(vectordb: Chroma) -> int:
    try:
        return vectordb._collection.count()
    except Exception:
        return 0
