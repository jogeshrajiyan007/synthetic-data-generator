def get_token():

    headersList = {"Content-Type": "application/x-www-form-urlencoded"}
    payload = f"grant_type=client_credentials&client_id={client_id}&client_secret={client_secret}&scope=urn:grp:chatgpt"
    response = requests.request(method='POST', url=url, data=payload, headers=headersList)
    if response.status_code != 200:
        raise Exception(f'Request failed with status {response.status_code}, {response.text}')
    token_info = response.json()
    access_token = token_info['access_token']
    return access_token

def query_llm(user_prompt, gpt4o_llm):
    try:
        prompt = ChatPromptTemplate.from_messages(
            [
                (
                    "system",
                    "You are a helpful assistant.",
                ),
                ("human", "{input}"),
            ]
        )
 
        chain = prompt | gpt4o_llm
        llm_response = chain.invoke(
            {
                "input": user_prompt,
            }
        )

        response_json = {
            "content": llm_response.content,
            "model": llm_response.response_metadata.get("model_name"),
            "model_provider": llm_response.response_metadata.get("model_provider"),
            "finish_reason": llm_response.response_metadata.get("finish_reason"),
            "run_id": llm_response.id,
            "usage": {
                "prompt_tokens": llm_response.usage_metadata.get("input_tokens"),
                "completion_tokens": llm_response.usage_metadata.get("output_tokens"),
                "total_tokens": llm_response.usage_metadata.get("total_tokens"),
            },
            "content_filter_results": llm_response.response_metadata.get("content_filter_results"),
            "tool_calls": llm_response.tool_calls,
            "additional_kwargs": llm_response.additional_kwargs,
        }
        return response_json
        
    except json.JSONDecodeError:
        print("Response is not valid JSON:", llm_response)
        return {"error": "Invalid JSON response from the model."}
    except Exception as e:
        print(str(e))
        return f"Error while calling LLM: {str(e)}"

from langchain_community.document_loaders import (
PyPDFLoader,
TextLoader,
UnstructuredWordDocumentLoader,
CSVLoader
)

def load_document(file_path):
    if file_path.endswith(".pdf"):
        loader = PyPDFLoader(file_path)
    elif file_path.endswith(".docx"):
        loader = UnstructuredWordDocumentLoader(file_path)
    elif file_path.endswith(".csv"):
        loader = CSVLoader(file_path)
    elif file_path.endswith(".txt"):
        loader = TextLoader(file_path)
    else:
        raise ValueError("Unsupported file type")

    return loader.load()

from langchain_text_splitters import RecursiveCharacterTextSplitter
import uuid
import re

def extract_advanced_metadata(chunk, file_name, doc_type="policy"):
    text = chunk.page_content.lower()

    keywords = list(set(re.findall(r"\b\w+\b", text)))[:5]

    section = "Unknown"

    lines = chunk.page_content.split("\n")
    if len(lines) > 0:
        section = lines[0][:100]

    sensitive_words = ["ssn", "aadhar", "password", "bank"]

    contains_pii = any(word in text for word in sensitive_words)

    sensitivity = "Confidential" if contains_pii else "Internal"

    return {
        "doc_id": str(uuid.uuid4()),
        "source": file_name,
        "file_type": file_name.split(".")[-1],
        "chunk_id": None,
        "section": section,
        "heading_level": "H2",
        "department": "Claims",
        "owner": "DAAS DMG",
        "sensitivity": sensitivity,
        "contains_pii": contains_pii,
        "keywords": keywords,
        "summary": chunk.page_content[:150],
        "created_date": "2023-01-01",
        "ingested_at": datetime.now().isoformat(),
        "language": "en"
    }


def split_documents(documents, file_name):
    splitter = RecursiveCharacterTextSplitter(chunk_size=800, chunk_overlap=100)
    chunks = splitter.split_documents(documents)
    enriched_chunks = []

    for i,chunk in enumerate(chunks):
        metadata = extract_advanced_metadata(chunk,file_name)
        metadata["chunk_id"] = i
        metadata.update(chunk.metadata)
        chunk.metadata = metadata
        enriched_chunks.append(chunk)
    return enriched_chunks

from langchain_community.vectorstores import Chroma

def create_vector_store(chunks):
    embedding = get_embeddings()

    vectordb = Chroma.from_documents(
        documents = chunks,
        embedding = embedding,
        persist_directory = "./chroma_db"
    )

    vectordb.persist()
    return vectordb
