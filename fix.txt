    "\"\"\"\n",
    "Dynamic SQLite logger.\n",
    "- Schema is derived from the data dict at runtime — no hardcoded column lists.\n",
    "- Tables are created once; new keys in the dict are added as new columns automatically.\n",
    "- Two dedicated tables: llm_responses | document_chunks\n",
    "\"\"\"\n",
    "\n",
    "def _get_conn() -> sqlite3.Connection:\n",
    "    conn = sqlite3.connect(SQLITE_DB, timeout=30)\n",
    "    conn.execute(\"PRAGMA journal_mode=WAL\")\n",
    "    conn.execute(\"PRAGMA busy_timeout=30000\")\n",
    "    return conn\n",
    "\n",
    "\n",
    "def _ensure_table(cursor: sqlite3.Cursor, table: str, record: dict) -> None:\n",
    "    \"\"\"Create table if absent; add any missing columns dynamically.\"\"\"\n",
    "    cols = \", \".join(f\"{k} TEXT\" for k in record)\n",
    "    cursor.execute(f\"CREATE TABLE IF NOT EXISTS {table} ({cols})\")\n",
    "\n",
    "    # Add columns that exist in record but not yet in table\n",
    "    cursor.execute(f\"PRAGMA table_info({table})\")\n",
    "    existing = {row[1] for row in cursor.fetchall()}\n",
    "    for key in record:\n",
    "        if key not in existing:\n",
    "            cursor.execute(f\"ALTER TABLE {table} ADD COLUMN {key} TEXT\")\n",
    "\n",
    "\n",
    "def db_insert(table: str, record: dict) -> None:\n",
    "    \"\"\"Insert a flat dict into a table. Schema is inferred dynamically.\"\"\"\n",
    "    # Flatten any nested dicts / lists to strings\n",
    "    flat = {\n",
    "        k: (json.dumps(v) if isinstance(v, (dict, list)) else str(v) if v is not None else None)\n",
    "        for k, v in record.items()\n",
    "    }\n",
    "    with _get_conn() as conn:\n",
    "        cursor = conn.cursor()\n",
    "        _ensure_table(cursor, table, flat)\n",
    "        placeholders = \", \".join(\"?\" for _ in flat)\n",
    "        cols         = \", \".join(flat.keys())\n",
    "        cursor.execute(\n",
    "            f\"INSERT INTO {table} ({cols}) VALUES ({placeholders})\",\n",
    "            list(flat.values())\n",
    "        )\n",
    "        conn.commit()\n",
    "\n",
    "\n",
    "def db_query(sql: str, params: tuple = ()) -> list:\n",
    "    \"\"\"Run a SELECT and return rows as a list of dicts.\"\"\"\n",
    "    with _get_conn() as conn:\n",
    "        conn.row_factory = sqlite3.Row\n",
    "        cursor = conn.cursor()\n",
    "        cursor.execute(sql, params)\n",
    "        return [dict(row) for row in cursor.fetchall()]\n",
    "\n",
    "\n",
    "print(\"✅ Dynamic SQLite logger ready.\")"
